dbo.spKinderVonVaeternLoesen

Description: Mit der Stored procedure können Varkombi -Kinder vom Vater gelöst werden um als eigenständige Artikel geführt zu werden.

CREATE PROCEDURE spKinderVonVaeternLoesen
	@kindArtikel TYPE_spKinderVonVaeternLoesen READONLY 
AS
BEGIN
	DELETE dbo.tEigenschaftKombiWert 
	WHERE tEigenschaftKombiWert.kEigenschaftKombi IN 
	(
		SELECT tArtikel.kEigenschaftKombi 
		FROM dbo.tArtikel 
		WHERE tArtikel.kArtikel IN 
		(
			SELECT kArtikel 
			FROM @kindArtikel
		)
	);

	DECLARE @vaterArtikelKeys Table(kArtikel int)
	Insert @vaterArtikelKeys 
	SELECT kVaterArtikel 
	FROM dbo.tArtikel
	WHERE kArtikel IN 
	(
		SELECT tArtikel.kArtikel 
		FROM dbo.tArtikel 
		WHERE tArtikel.kArtikel IN 
		(
			SELECT kArtikel 
			FROM @kindArtikel
		)
	);

	UPDATE dbo.tArtikel 
	SET tArtikel.kVaterArtikel = 0, 
		tArtikel.kEigenschaftKombi = 0
	WHERE kArtikel IN 
	(
		SELECT tArtikel.kArtikel 
		FROM dbo.tArtikel 
		WHERE tArtikel.kArtikel IN 
		(
			SELECT kArtikel 
			FROM @kindArtikel
		)
	);
	
	UPDATE dbo.tArtikel 
	SET nIstVater = 0 
	WHERE  kArtikel IN 
	(
		SELECT kArtikel 
		FROM @vaterArtikelKeys
	)
	AND NOT EXISTS
	(
		SELECT kArtikel 
		FROM dbo.tArtikel 
		WHERE kVaterArtikel IN 
		(
			SELECT kArtikel 
			FROM @vaterArtikelKeys
		)
	);
END
