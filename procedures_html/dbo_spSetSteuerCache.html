CREATE PROCEDURE dbo.spSetSteuerCache
AS
BEGIN
	SET NOCOUNT ON;
	SET ANSI_NULLS ON;
	SET XACT_ABORT ON;

	DECLARE @retry INT = 5;
	WHILE @retry > 0
	BEGIN
		BEGIN TRY
			DECLARE @TranCount INT = @@TRANCOUNT;
			IF (@TranCount = 0)
			BEGIN
				BEGIN TRANSACTION;
			END;

			TRUNCATE TABLE dbo.tSteuercache;

			INSERT INTO dbo.tSteuercache (kSteuerklasse, fSteuersatz, kFirma, kSteuerzone)
			SELECT DISTINCT
				vSteuercache.kSteuerKlasse,
				vSteuercache.fSteuersatz,
				vSteuercache.kFirma,
				vSteuercache.kSteuerZone
			FROM dbo.vSteuercache;

			SET @retry = 0;
			IF (@TranCount = 0)
			BEGIN
				COMMIT TRANSACTION;
			END;
		END TRY
		BEGIN CATCH
			IF (ERROR_NUMBER() = 1205 AND @TranCount = 0 AND XACT_STATE() <> 0)
			BEGIN
				SET @retry = @retry - 1;
				ROLLBACK;
			END;
			ELSE
			BEGIN
				SET @retry = 0;
				THROW;
			END;
		END CATCH;
	END;
END;
