CREATE PROCEDURE [FulfillmentNetwork].[spAbrechnungLagerplatzErstellenInitPreview]
    @kLieferant AS INT ,
    @kBenutzer AS INT ,
    @cSessionId AS [VARCHAR](255)
AS
    EXEC FulfillmentNetwork.spAbrechnungLagerplatzErstellenEndPreview @cSessionId = @cSessionId;
	
    INSERT  INTO FulfillmentNetwork.tAbrechnungLagerplatzPreview
            ( kLieferant ,
              kWarenLagerPlatz ,
              kWarenLagerPlatzTyp ,
              kWMSLagerBereich ,
              kWarenlager ,
              nModus ,
              nLagerkostenBerechnung ,
              kArtikel ,
              kBenutzer ,
              cSessionId ,
              dZeitstempel
	        )
            SELECT  kLieferant ,
                    kWarenLagerPlatz ,
                    kWarenLagerPlatzTyp ,
                    kWMSLagerBereich ,
                    kWarenlager ,
                    nModus ,
                    nLagerkostenBerechnung ,
                    kArtikel ,
                    @kBenutzer AS kBenutzer ,
                    @cSessionId AS cSessionId ,
                    GETDATE() AS dZeitstempel
            FROM    FulfillmentNetwork.tAbrechnungLagerplatz
            WHERE   kLieferant = @kLieferant;
