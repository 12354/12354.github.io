CREATE PROCEDURE [Auslieferung].[spPicklistenAuftraegeEntfernen]  

-- Format:
-- <Bestellung>
--   <kBestellung>1</kBestellung>
-- </Bestellung>

@Auftraege XML,

-- Enthält den Benutzer zur eindeutigen Zuordnung
@kBenutzer INT,

-- Sitzungsid
@kSessionId INT

AS  
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
  
-- 
-- XML-Geraffel umkopieren
-- 
IF(OBJECT_ID('tempdb..#Bestellungen') IS NOT NULL)
BEGIN
	DROP TABLE #Bestellungen;
END
CREATE TABLE #Bestellungen(
	kBestellung INT
)
INSERT INTO #Bestellungen
(
	#Bestellungen.kBestellung
)
SELECT
	ParamValues.ID.value('kBestellung[1]', 'INT') AS kBestellung
FROM @Auftraege.nodes('/Bestellung') AS ParamValues(ID)

DELETE dbo.tPicklistePos
FROM dbo.tPicklistePos 
JOIN dbo.tPickliste ON dbo.tPickliste.kPickliste = dbo.tPicklistePos.kPickliste AND dbo.tPickliste.kSessionId = @kSessionId
JOIN Verkauf.vAuftragPosition ON vAuftragPosition.kAuftragPosition = dbo.tPicklistePos.kBestellPos
JOIN #Bestellungen ON #Bestellungen.kBestellung = vAuftragPosition.kAuftrag
