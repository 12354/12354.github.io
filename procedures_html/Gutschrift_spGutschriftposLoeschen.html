Gutschrift.spGutschriftposLoeschen

CREATE PROCEDURE [Gutschrift].[spGutschriftposLoeschen]
	@xGutschriftpos XML = NULL,
	@kGutschriftpos INT = NULL
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @retry INT;
SET @retry = 5;
WHILE @retry > 0
BEGIN
	SET CONTEXT_INFO 0x5119;
	BEGIN TRY
		BEGIN TRANSACTION
			IF(OBJECT_ID('tempdb..#Gutschriftpos') IS NOT NULL)
			BEGIN
				DROP TABLE #Gutschriftpos;
			END
			CREATE TABLE #Gutschriftpos(kGutschriftpos INT, kRechnungPosition INT);
			IF(@xGutschriftpos IS NOT NULL)
			BEGIN
				INSERT INTO #Gutschriftpos(kGutschriftpos, kRechnungPosition)
					SELECT Gutschriftpos.ID.value('kGutschriftPos[1]', 'INT'),
							dbo.tgutschriftpos.kBestellPos
						FROM @xGutschriftpos.nodes('Gutschriftpos') AS Gutschriftpos(ID)
						JOIN dbo.tgutschriftpos ON dbo.tgutschriftpos.kGutschriftPos = Gutschriftpos.ID.value('kGutschriftPos[1]', 'INT');
			END
			ELSE
			BEGIN
				DECLARE @kRechnungPosition INT;
				SELECT @kRechnungPosition = kBestellpos
					FROM dbo.tgutschriftpos 
					WHERE dbo.tgutschriftpos.kGutschriftPos = @kGutschriftpos;
				INSERT INTO #Gutschriftpos(kGutschriftpos, kRechnungPosition)
					VALUES (@kGutschriftpos, @kRechnungPosition);
			END			
			DELETE FROM dbo.tGutschriftpos
				WHERE dbo.tGutschriftpos.kGutschriftPos IN (SELECT #Gutschriftpos.kGutschriftpos FROM #Gutschriftpos);
			IF(EXISTS(SELECT * FROM #Gutschriftpos WHERE #Gutschriftpos.kRechnungPosition IS NOT NULL))
			BEGIN
				DECLARE @Auftraege AS Verkauf.TYPE_spAuftragEckdatenBerechnen;
				
				INSERT INTO @Auftraege(kAuftrag)
				SELECT DISTINCT vAuftrag.kAuftrag
				FROM Verkauf.vAuftrag
				JOIN Rechnung.vRechnungPosition ON vRechnungPosition.kAuftrag = vAuftrag.kAuftrag
				JOIN #Gutschriftpos ON #Gutschriftpos.kRechnungPosition = vRechnungPosition.kRechnungPosition;

				EXEC Verkauf.spAuftragEckdatenBerechnen @Auftrag = @Auftraege;

			END
			SET @retry = -1;
			SET CONTEXT_INFO 0x0;
		COMMIT
	END TRY
	BEGIN CATCH
	IF(ERROR_NUMBER() = 1205)
			BEGIN
				SET @retry = @retry - 1;
				ROLLBACK;
				IF(@retry = 0)
				BEGIN
					SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
					RAISERROR (	@ErrorMessage, 
								@ErrorSeverity,
								@ErrorState
					);
					SET CONTEXT_INFO 0x0;
					RETURN;
				END
				WAITFOR DELAY '00:00:00:5';
			END
			ELSE 
			BEGIN
				SET @retry = -1;
				SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
				ROLLBACK;
				RAISERROR (	@ErrorMessage, 
							@ErrorSeverity,
							@ErrorState
				);
				SET CONTEXT_INFO 0x0;
				RETURN;
			END
	END CATCH
	SET CONTEXT_INFO 0x0;	
END
