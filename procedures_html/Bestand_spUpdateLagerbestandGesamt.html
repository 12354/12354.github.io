CREATE PROCEDURE Bestand.spUpdateLagerbestandGesamt
    @Artikel AS dbo.TYPE_spUpdateLagerbestand READONLY, -- Liste von Artikeln
	@Debug BIT = 0 -- = 1, um Debug
AS
/*************************************************************************************************************
Beschreibung:	Berechnet für mehrere Artikel sowohl Lagerbestand als auch LagerbestandProLager neu
--------------------------------------------------------------------------------------------------------------
-- Beispielaufruf:
DECLARE @Artikel dbo.TYPE_spUpdateLagerbestand;
INSERT INTO @Artikel (kArtikel) SELECT TOP(100) kArtikel FROM dbo.tArtikel ORDER BY kArtikel DESC;
EXEC Bestand.spUpdateLagerbestandGesamt @Artikel = @Artikel;
*************************************************************************************************************/

BEGIN
	SET NOCOUNT, ANSI_NULLS, ANSI_NULL_DFLT_ON, ANSI_PADDING, CONCAT_NULL_YIELDS_NULL, XACT_ABORT ON;

	DECLARE @ArtikelUndWarenlager dbo.TYPE_spUpdateLagerbestandProLager;

	-- Lagerbestand neu berechnen
	EXEC dbo.spUpdateLagerbestand
		@Artikel = @Artikel,
		@Debug = @Debug;

	-- LagerbestandProLager neu berechnen
	INSERT INTO @ArtikelUndWarenlager (kArtikel, kWarenlager)
	SELECT		Input.kArtikel, WL.kWarenlager
	FROM		@Artikel AS Input
	CROSS JOIN	dbo.tWarenlager AS WL

	EXEC dbo.spUpdateLagerbestandProLager @Artikel = @ArtikelUndWarenlager
END
