CREATE PROCEDURE [Lieferantenbestellung].[spLieferantenBestellungLoeschen]
		@xLieferantenbestellung XML = NULL, 
		@kLieferantenbestellung INT = NULL -- Noch TODO
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @retry INT;
SET @retry = 5;
WHILE @retry > 0
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION
			IF(OBJECT_ID('tempdb..#Lieferantenbestellung') IS NOT NULL)
			BEGIN
				DROP TABLE #Lieferantenbestellung;
			END
			CREATE TABLE #Lieferantenbestellung(
				kLieferantenbestellung INT,
				kLieferantenBestellungLA INT,
				kLieferantenBestellungRA INT,
				kLieferantenBestellungLieferant INT
			);
			IF(@xLieferantenBestellung IS NOT NULL)
			BEGIN
				INSERT INTO #Lieferantenbestellung(kLieferantenbestellung, kLieferantenBestellungLA, kLieferantenBestellungRA, kLieferantenBestellungLieferant)
					SELECT	Lieferantenbestellung.ID.value('kLieferantenbestellung[1]', 'INT'),
							dbo.tLieferantenBestellungLA.kLieferantenBestellungLA,
							dbo.tLieferantenBestellungRA.kLieferantenBestellungRA,
							dbo.tLieferantenBestellungLieferant.kLieferantenBestellungLieferant
						FROM @xLieferantenbestellung.nodes('/Lieferantenbestellung') AS Lieferantenbestellung(ID)
						JOIN dbo.tLieferantenBestellung ON Lieferantenbestellung.ID.value('kLieferantenbestellung[1]', 'INT') = dbo.tLieferantenBestellung.kLieferantenBestellung
						LEFT JOIN dbo.tLieferantenBestellungLA ON dbo.tLieferantenBestellungLA.kLieferantenBestellungLA = dbo.tLieferantenBestellung.kLieferantenBestellungLA
						LEFT JOIN dbo.tLieferantenBestellungRA ON dbo.tLieferantenBestellungRA.kLieferantenBestellungRA = dbo.tLieferantenBestellung.kLieferantenBestellungRA
						LEFT JOIN dbo.tLieferantenBestellungLieferant ON dbo.tLieferantenBestellungLieferant.kLieferantenBestellungLieferant = dbo.tLieferantenBestellung.kLieferantenBestellungLieferant;
			END
			ELSE
			BEGIN
				INSERT INTO #Lieferantenbestellung(kLieferantenbestellung, kLieferantenBestellungLA, kLieferantenBestellungRA, kLieferantenBestellungLieferant)
					SELECT @kLieferantenBestellung,
							dbo.tLieferantenBestellungLA.kLieferantenBestellungLA,
							dbo.tLieferantenBestellungRA.kLieferantenBestellungRA,
							dbo.tLieferantenBestellungLieferant.kLieferantenBestellungLieferant 						
						FROM dbo.tLieferantenBestellung
						LEFT JOIN dbo.tLieferantenBestellungLA ON dbo.tLieferantenBestellungLA.kLieferantenBestellungLA = dbo.tLieferantenBestellung.kLieferantenBestellungLA
						LEFT JOIN dbo.tLieferantenBestellungRA ON dbo.tLieferantenBestellungRA.kLieferantenBestellungRA = dbo.tLieferantenBestellung.kLieferantenBestellungRA
						LEFT JOIN dbo.tLieferantenBestellungLieferant ON dbo.tLieferantenBestellungLieferant.kLieferantenBestellungLieferant = dbo.tLieferantenBestellung.kLieferantenBestellungLieferant
						WHERE dbo.tLieferantenBestellung.kLieferantenbestellung = @kLieferantenbestellung;
			END			
			
			DECLARE @xLieferantenbestellungPos XML;
			SELECT @xLieferantenbestellungPos = (
					SELECT dbo.tLieferantenBestellungPos.kLieferantenBestellungPos,
							dbo.tLieferantenBestellungPos.kArtikel,
							dbo.tLieferantenBestellung.nStatus,
							dbo.tLieferantenBestellungPos.fMengeGeliefert
						FROM #Lieferantenbestellung
						JOIN dbo.tLieferantenBestellungPos ON #Lieferantenbestellung.kLieferantenbestellung = dbo.tLieferantenBestellungPos.kLieferantenBestellung 
						JOIN dbo.tlieferantenBestellung ON dbo.tlieferantenBestellungPos.kLieferantenBestellung = dbo.tlieferantenbestellung.kLieferantenBestellung
						FOR XML PATH('LieferantenbestellungPos'), TYPE
				);
			EXEC Lieferantenbestellung.spLieferantenBestellungPosLoeschen @xLieferantenbestellungPos;
			
			DELETE dbo.tLieferantenBestellungRA
				FROM dbo.tLieferantenBestellungRA 
				JOIN dbo.tLieferantenBestellung ON dbo.tLieferantenBestellungRA.kLieferantenBestellungRA = dbo.tLieferantenBestellung.kLieferantenBestellungRA
				JOIN #Lieferantenbestellung ON #Lieferantenbestellung.kLieferantenbestellung = dbo.tLieferantenBestellung.kLieferantenBestellung
				WHERE #Lieferantenbestellung.kLieferantenBestellungRA NOT IN(
						SELECT LB.kLieferantenBestellungRA
							FROM dbo.tLieferantenBestellung AS LB
							WHERE LB.kLieferantenBestellung NOT IN(SELECT LBTemp.kLieferantenBestellung FROM #Lieferantenbestellung AS LBTemp));

			DELETE dbo.tLieferantenBestellungLA
				FROM dbo.tLieferantenBestellungLA 
				JOIN dbo.tLieferantenBestellung ON dbo.tLieferantenBestellungLA.kLieferantenBestellungLA = dbo.tLieferantenBestellung.kLieferantenBestellungLA
				JOIN #Lieferantenbestellung ON #Lieferantenbestellung.kLieferantenbestellung = dbo.tLieferantenBestellung.kLieferantenBestellung
				WHERE #Lieferantenbestellung.kLieferantenBestellungLA NOT IN(
						SELECT LB.kLieferantenBestellungLA
							FROM dbo.tLieferantenBestellung AS LB
							WHERE LB.kLieferantenBestellung NOT IN(SELECT LBTemp.kLieferantenbestellung FROM #Lieferantenbestellung AS LBTemp));
			
			DELETE dbo.tLieferantenBestellungLieferant
				FROM dbo.tLieferantenBestellungLieferant 
				JOIN dbo.tLieferantenBestellung ON dbo.tLieferantenBestellung.kLieferantenBestellungLieferant = dbo.tLieferantenBestellungLieferant.kLieferantenBestellungLieferant
				JOIN #Lieferantenbestellung ON #Lieferantenbestellung.kLieferantenbestellung = dbo.tLieferantenBestellung.kLieferantenBestellung
				WHERE #Lieferantenbestellung.kLieferantenBestellungLieferant NOT IN(
						SELECT LB.kLieferantenBestellungLieferant
							FROM dbo.tLieferantenBestellung AS LB
							WHERE LB.kLieferantenBestellung NOT IN(SELECT LBTemp.kLieferantenbestellung FROM #Lieferantenbestellung  AS LBTemp));
			SET CONTEXT_INFO 0x5122;
			DELETE dbo.tLieferantenBestellung
				FROM dbo.tLieferantenBestellung 
				JOIN #Lieferantenbestellung ON #Lieferantenbestellung.kLieferantenbestellung = dbo.tLieferantenBestellung.kLieferantenBestellung;
			SET @retry = -1;
			SET CONTEXT_INFO 0x0;
		COMMIT
	END TRY
	BEGIN CATCH
	IF(ERROR_NUMBER() = 1205)
			BEGIN
				SET @retry = @retry - 1;
				ROLLBACK;
				IF(@retry = 0)
				BEGIN
					SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
					RAISERROR (	@ErrorMessage, 
								@ErrorSeverity,
								@ErrorState
					);
					SET CONTEXT_INFO 0x0;
					RETURN;
				END
				WAITFOR DELAY '00:00:00:5';
			END
			ELSE 
			BEGIN
				SET @retry = -1;
				SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
				ROLLBACK;
				RAISERROR (	@ErrorMessage, 
							@ErrorSeverity,
							@ErrorState
				);
				SET CONTEXT_INFO 0x0;
				RETURN;
			END
	END CATCH
	SET CONTEXT_INFO 0x0;		
END
