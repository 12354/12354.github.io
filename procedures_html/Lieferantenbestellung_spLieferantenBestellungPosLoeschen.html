Lieferantenbestellung.spLieferantenBestellungPosLoeschen

Description: Mit der Stored Procedure können Einträge in der Tabelle dbo.tLieferantenBestellungPos gelöscht werden.

CREATE PROCEDURE [Lieferantenbestellung].[spLieferantenBestellungPosLoeschen]
	@xLieferantenbestellungPos XML = NULL,
	@kLieferantenbestellungPos INT = NULL
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @retry INT;
SET @retry = 5;
WHILE @retry > 0
BEGIN
	SET CONTEXT_INFO 0x5125;
	BEGIN TRY
		BEGIN TRANSACTION
			IF(OBJECT_ID('tempdb..#LieferantenBestellungPos') IS NOT NULL)
			BEGIN
				DROP TABLE #LieferantenBestellungPos;
			END
			CREATE TABLE #LieferantenBestellungPos(
					kLieferantenbestellungPos INT,
					kArtikel INT,
					nStatus INT,
					fMengeGeliefert DECIMAL(25,13),
					fMenge DECIMAL(25,13)
					);
			IF(@xLieferantenbestellungPos IS NOT NULL)
			BEGIN
				INSERT INTO #LieferantenBestellungPos(kLieferantenbestellungPos, kArtikel, nStatus, fMengeGeliefert,fMenge)
					SELECT	LieferantenbestellungPos.ID.value('kLieferantenBestellungPos[1]', 'INT'),
							dbo.tLieferantenBestellungPos.kArtikel,
							dbo.tLieferantenBestellung.nStatus,
							dbo.tLieferantenBestellungPos.fMengeGeliefert,
							dbo.tLieferantenBestellungPos.fMenge
						FROM @xLieferantenbestellungPos.nodes('LieferantenbestellungPos') AS LieferantenbestellungPos(ID)
						JOIN dbo.tLieferantenBestellungPos ON dbo.tLieferantenBestellungPos.kLieferantenBestellungPos = LieferantenbestellungPos.ID.value('kLieferantenBestellungPos[1]', 'INT')
						JOIN dbo.tLieferantenBestellung ON dbo.tLieferantenBestellung.kLieferantenBestellung = dbo.tLieferantenBestellungPos.kLieferantenBestellung;
			END 
			ELSE
			BEGIN
				INSERT INTO #LieferantenBestellungPos(kLieferantenbestellungPos, kArtikel, nStatus, fMengeGeliefert,fMenge)
					SELECT	@kLieferantenbestellungPos,
							dbo.tLieferantenBestellungPos.kArtikel,
							dbo.tLieferantenBestellung.nStatus,
							dbo.tLieferantenBestellungPos.fMengeGeliefert,
							dbo.tLieferantenBestellungPos.fMenge
						FROM dbo.tLieferantenBestellungPos 
						JOIN dbo.tLieferantenBestellung ON dbo.tLieferantenBestellung.kLieferantenBestellung = dbo.tLieferantenBestellungPos.kLieferantenBestellung
						WHERE dbo.tLieferantenBestellungpos.kLieferantenbestellungpos = @kLieferantenbestellungPos;
			END
			DELETE FROM dbo.tLieferantenBestellungPos
				WHERE tLieferantenBestellungPos.kLieferantenBestellungPos IN (SELECT #LieferantenBestellungPos.kLieferantenBestellungPos FROM #LieferantenBestellungPos );			
			IF(EXISTS(SELECT * FROM #LieferantenBestellungPos 
						WHERE (nStatus > 5 AND nStatus < 50)
							AND (fMenge - fMengeGeliefert)  > 0))
			BEGIN
				DECLARE @Lagerbestand TYPE_spUpdateLagerbestand;
				INSERT INTO @Lagerbestand ( kArtikel )
				SELECT DISTINCT kArtikel 
				FROM #LieferantenBestellungPos;

				EXEC dbo.spUpdateLagerbestand @Artikel = @Lagerbestand;
			END
			SET @retry = -1;
			SET CONTEXT_INFO 0x0;
		COMMIT
	END TRY
	BEGIN CATCH
	IF(ERROR_NUMBER() = 1205)
			BEGIN
				SET @retry = @retry - 1;
				ROLLBACK;
				IF(@retry = 0)
				BEGIN
					SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
					RAISERROR (	@ErrorMessage, 
								@ErrorSeverity,
								@ErrorState
					);
					SET CONTEXT_INFO 0x0;
					RETURN;
				END
				WAITFOR DELAY '00:00:00:5';
			END
			ELSE 
			BEGIN
				SET @retry = -1;
				SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
				ROLLBACK;
				RAISERROR (	@ErrorMessage, 
							@ErrorSeverity,
							@ErrorState
				);
				SET CONTEXT_INFO 0x0;
				RETURN;
			END
	END CATCH
	SET CONTEXT_INFO 0x0;	
END
