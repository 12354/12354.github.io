CREATE PROCEDURE [Kunde].[spGetNextDebitorennr]
@nextNumber Int = 0 OUTPUT
AS
BEGIN


SELECT @nextNumber = 
(
	SELECT TOP 1 * FROM 
	(
		SELECT t1.nDebitorennr+1 AS nDebitorennr
		FROM tkunde t1
		WHERE NOT EXISTS
		(
			SELECT * FROM tkunde t2 WHERE t2.nDebitorennr = t1.nDebitorennr + 1
		)
		AND  t1.nDebitorennr+1 > (SELECT CONVERT(INT,cValue) + 1 FROM tOptions WHERE cKey = 'nDebitorenNr')
		AND t1.nDebitorennr IS NOT NULL
		AND t1.nDebitorennr > 10000
		UNION --Wenn noch keine vergeben
		SELECT CONVERT(INT,cValue) + 1 
		FROM tOptions WHERE cKey = 'nDebitorenNr'
		AND NOT EXISTS 
		(
			SELECT * FROM tkunde t3 WHERE t3.nDebitorennr = CONVERT(INT,cValue) + 1 
		)
	) ot
	ORDER BY 1
)
UPDATE tOptions SET cValue = @nextNumber WHERE cKey = 'nDebitorenNr'

END
