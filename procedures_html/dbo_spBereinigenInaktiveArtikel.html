dbo.spBereinigenInaktiveArtikel

Description: Die Procedure löscht alle inaktiven Artikel dauerhaft aus der Datenbank.

CREATE PROCEDURE [dbo].[spBereinigenInaktiveArtikel]
AS
     SET NOCOUNT OFF;
     SET ANSI_NULLS ON;
     SET ANSI_NULL_DFLT_ON ON;
     SET ANSI_PADDING ON;
     SET CONCAT_NULL_YIELDS_NULL ON;
     SET XACT_ABORT OFF;
    BEGIN
        DELETE item
        FROM dbo.tArtikel AS item
        -- TODO: Auftrag, Warenlagereingang, Retouren, Lieferantenbestellung -> Sollte der Artikel dort in Verwendung sein, darf er nicht mehr gelöscht werden
        WHERE item.cAktiv = 'N'
              AND NOT EXISTS
        (
            SELECT 1
            FROM Artikel.tArtikelType AS ArtT
            WHERE item.kArtikel = ArtT.kArtikel
                  AND ArtT.uArtikelType IN('0200D767-9A5C-4B6A-882E-051FFA69A9D2', 'B268BC2F-55D4-468D-B55A-A6317DD78D0E')
        );
    END;
