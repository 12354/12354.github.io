CREATE PROCEDURE [Pos].[spUpdateEntityTrackingArtikelSonderpreise]
	@kShop INT
AS
SET NOCOUNT ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET ANSI_WARNINGS ON;
SET ANSI_PADDING ON;
BEGIN
	UPDATE Sync.tEntityTracking
		SET tEntityTracking.dLastModified = GETDATE()
	FROM Sync.tEntityTracking
	JOIN dbo.tArtikelSonderpreis ON tArtikelSonderpreis.kArtikel = tEntityTracking.kEntityId
	JOIN Sync.tEntityPlatform ON tEntityPlatform.kEntityId = tEntityTracking.kEntityId
		AND tEntityPlatform.nEntityType = tEntityTracking.nEntityType
		AND tEntityTracking.nSyncType = tEntityTracking.nSyncType
	LEFT JOIN dbo.tlagerbestand ON tlagerbestand.kArtikel = tArtikelSonderpreis.kArtikel
	WHERE (CAST(tEntityTracking.dLastModified AS DATE) < CAST(tArtikelSonderpreis.dStart AS DATE) AND CAST(tArtikelSonderpreis.dStart AS DATE) <= CAST(GETDATE() AS DATE)
		OR CAST(tEntityTracking.dLastModified AS DATE) < DATEADD(DAY, 1, CAST(tArtikelSonderpreis.dEnde AS DATE)) AND CAST(tArtikelSonderpreis.dEnde AS DATE) < CAST(GETDATE() AS DATE))
		AND tArtikelSonderpreis.nAktiv = 1
		AND tEntityTracking.nEntityType = 1
		AND tEntityTracking.nSyncType = 4
		AND tEntityPlatform.kShop = @kShop
END
