CREATE PROCEDURE dbo.spErrorHandler
    @Context NVARCHAR(128) = NULL,
    @DeadlockRetries TINYINT OUTPUT
AS
BEGIN
	SET NOCOUNT, ANSI_NULLS, ANSI_NULL_DFLT_ON, ANSI_PADDING, CONCAT_NULL_YIELDS_NULL, XACT_ABORT ON;

    -- Auskommentiert, da @CurrentLocks wird aktuell nicht gespeichert wird
	-- DECLARE @CurrentLocks XML = (SELECT * FROM Maintenance.vLocks ORDER BY SPID FOR XML PATH('TransactionLock'));
    DECLARE @DeadlockGraph XML = NULL;

    -- 13 -> Indicates transaction deadlock errors | 1205 = Deadlock
	IF ((ERROR_SEVERITY() = 13) OR (ERROR_NUMBER() = 1205))
    BEGIN
        SELECT TOP(1) @DeadlockGraph = DeadlockGraph FROM Maintenance.vLastDeadlockGraphs ORDER BY [Timestamp] DESC;

		-- Extrem große Deadlock-Graphen aus Speicherplatz-Gründen nicht speichern
		IF LEN(CONVERT(NVARCHAR(MAX), @DeadlockGraph)) > 50000
		BEGIN
			SET @DeadlockGraph = NULL;
		END

    END

	IF (XACT_STATE() = -1) OR ((ISNULL(@DeadlockRetries, 0) > 0) AND (@@TRANCOUNT > 0) AND (XACT_STATE() <> 0)) ROLLBACK TRANSACTION;

    INSERT INTO	dbo.tDbErrorlog
				(dZeitstempel, cAppName, cHostName, cInformation,
				nErrorNumber, nErrorSeverity, nErrorLine, nErrorState, cErrorProcedure,
				cErrorMessage, cLockStatus, cDeadlockGraph)
	VALUES		(GETUTCDATE(), APP_NAME(), HOST_NAME(), @Context,
				ERROR_NUMBER(), ERROR_SEVERITY(), ERROR_LINE(), ERROR_STATE(), ERROR_PROCEDURE(),
				ERROR_MESSAGE(), NULL, @DeadlockGraph);

    IF (ISNULL(@DeadlockRetries, 0) > 1) AND (ERROR_NUMBER() IN (1203, 1205, 1221, 1532, 1533, 1534, 1535, 3948, 3953, 3957, 3960, 3966, 41301, 41339))
    BEGIN
        SET @DeadlockRetries = @DeadlockRetries - 1;
        DECLARE @time CHAR(8) = '00:00:0' + CAST(CAST(RAND() * 5 AS INT) + 1 AS CHAR(1));
        WAITFOR DELAY @time;
    END ELSE BEGIN
        SET @DeadlockRetries = 0;
    END

    RETURN @DeadlockRetries;
END
