CREATE PROCEDURE spGetKategorienRabattById
    @kategorieKeys TYPE_spGetKategorienRabattById READONLY
AS
BEGIN
    SET NOCOUNT ON;
 
    SELECT * 
	FROM dbo.tKategorieRabatt 
	WHERE tKategorieRabatt.kKategorie IN
    (
        SELECT kKategorien
        FROM @kategorieKeys
    );
END
