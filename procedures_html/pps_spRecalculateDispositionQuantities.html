CREATE PROCEDURE pps.spRecalculateDispositionQuantities
    @disposition AS pps.TYPE_spRecalculateDispositionQuantities READONLY
AS
BEGIN
    DECLARE @retry INT = 5;
    WHILE @retry > 0
    BEGIN
        BEGIN TRY
            DECLARE @TranCount INT = @@TRANCOUNT;
            IF (@TranCount = 0)
            BEGIN
                BEGIN TRANSACTION
            END

            -- Übergebene Einträge
            IF EXISTS (SELECT kDisposition FROM @disposition)
            BEGIN
                UPDATE pps.tDisposition
                SET -- Istmenge
                    tDisposition.fActualQuantity = ISNULL(actual.fActualQuantity, 0.00),
                    -- Ziel-/Planmenge aus Aufträgen (Beschaffung/Verkauf)
                    tDisposition.fPlannedQuantity = ISNULL(orders.fTargetQuantity, 0.00)
                                                    - ISNULL(orders.fActualQuantity, 0.00),
                    -- Status
                    tDisposition.nStatus = CASE
                                               WHEN ISNULL(actual.fActualQuantity, 0.00) >= tDisposition.fQuantity THEN
                                                   3                    -- vollständig produziert
                                               WHEN ISNULL(actual.fActualQuantity, 0.00) < tDisposition.fQuantity
                                                    AND ISNULL(actual.fActualQuantity, 0.00) > 0.0 THEN
                                                   4                    -- teilweise produziert
                                               WHEN ISNULL(orders.fTargetQuantity, 0.00)
                                                    - ISNULL(orders.fActualQuantity, 0.00) > 0 THEN
                                                   1                    -- nur auf Fertigungsauftrag
                                               WHEN ISNULL(orders.fActualQuantity, 0.00) = 0 THEN
                                                   tDisposition.nStatus -- unverändert (aktuellen Status beibehalten)
                                               ELSE
                                                   99                   -- unbekannt, daher "Pending"
                                           END
                FROM pps.tDisposition
                    JOIN @disposition AS dispo
                        ON tDisposition.kDisposition = dispo.kDisposition
                    LEFT JOIN
                    (
                        SELECT tWorkJobDisposition.kDisposition,
                               SUM(tWorkJob.fActualQuantity) AS fActualQuantity
                        FROM pps.tWorkJobDisposition
                            JOIN pps.tWorkJob
                                ON tWorkJob.kWorkJob = tWorkJobDisposition.kWorkJob
                        GROUP BY tWorkJobDisposition.kDisposition
                    ) AS actual
                        ON tDisposition.kDisposition = actual.kDisposition
                    LEFT JOIN
                    (
                        SELECT tProductionOrderProducible.kDisposition,
                               SUM(tProductionOrderProducible.fActualQuantity) AS fActualQuantity,
                               SUM(tProductionOrderProducible.fTargetTotalQuantity) AS fTargetQuantity
                        FROM pps.tProductionOrderProducible
                        GROUP BY tProductionOrderProducible.kDisposition
                    ) AS orders
                        ON tDisposition.kDisposition = orders.kDisposition
            END
            ELSE
            BEGIN
                -- Alle nicht erledigten
                UPDATE pps.tDisposition
                SET -- Istmenge
                    tDisposition.fActualQuantity = ISNULL(actual.fActualQuantity, 0.00),
                    -- Ziel-/Planmenge aus Aufträgen (Beschaffung/Verkauf)
                    tDisposition.fPlannedQuantity = ISNULL(orders.fTargetQuantity, 0.00)
                                                    - ISNULL(orders.fActualQuantity, 0.00),
                    -- Status
                    tDisposition.nStatus = CASE
                                               WHEN ISNULL(actual.fActualQuantity, 0.00) >= tDisposition.fQuantity THEN
                                                   3                    -- vollständig produziert
                                               WHEN ISNULL(actual.fActualQuantity, 0.00) < tDisposition.fQuantity
                                                    AND ISNULL(actual.fActualQuantity, 0.00) > 0.0 THEN
                                                   4                    -- teilweise produziert
                                               WHEN ISNULL(orders.fTargetQuantity, 0.00)
                                                    - ISNULL(orders.fActualQuantity, 0.00) > 0 THEN
                                                   1                    -- nur auf Fertigungsauftrag
                                               WHEN ISNULL(orders.fActualQuantity, 0.00) = 0 THEN
                                                   tDisposition.nStatus -- unverändert (aktuellen Status beibehalten)
                                               ELSE
                                                   99                   -- unbekannt, daher "Pending"
                                           END
                FROM pps.tDisposition
                    LEFT JOIN
                    (
                        SELECT tWorkJobDisposition.kDisposition,
                               SUM(tWorkJob.fActualQuantity) AS fActualQuantity
                        FROM pps.tWorkJobDisposition
                            JOIN pps.tWorkJob
                                ON tWorkJob.kWorkJob = tWorkJobDisposition.kWorkJob
                        GROUP BY tWorkJobDisposition.kDisposition
                    ) AS actual
                        ON tDisposition.kDisposition = actual.kDisposition
                    LEFT JOIN
                    (
                        SELECT tProductionOrderProducible.kDisposition,
                               SUM(tProductionOrderProducible.fActualQuantity) AS fActualQuantity,
                               SUM(tProductionOrderProducible.fTargetTotalQuantity) AS fTargetQuantity
                        FROM pps.tProductionOrderProducible
                        GROUP BY tProductionOrderProducible.kDisposition
                    ) AS orders
                        ON tDisposition.kDisposition = orders.kDisposition
                WHERE tDisposition.nStatus <> 3
            END
            SET @retry = 0;
            IF (@TranCount = 0)
            BEGIN
                COMMIT TRANSACTION
            END
        END TRY
        BEGIN CATCH
            IF (ERROR_NUMBER() = 1205 AND @TranCount = 0 AND XACT_STATE() <> 0)
            BEGIN
                SET @retry = @retry - 1;
                ROLLBACK
            END
            ELSE
            BEGIN
                SET @retry = 0;
                THROW;
            END
        END CATCH
    END
END
