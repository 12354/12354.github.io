CREATE PROCEDURE [Auslieferung].[spReserviereBestellungen]  

-- Reservierung mehrerer Bestellungen
--
-- Transaction-Handling: siehe https://msdn.microsoft.com/en-us/library/ms188378.aspx
--

-- <Bestellung>
--   <kBestellung>1</kBestellung>
--   <nPrio>0</nPrio>
-- </Bestellung>
-- <Bestellung>
--   <kBestellung>4</kBestellung>
--   <nPrio>1</nPrio>
-- </Bestellung>
-- <Bestellung>
--   <kBestellung>2</kBestellung>
--   <nPrio>2</nPrio>
-- </Bestellung>
@Bestellungen XML,

@Laeger XML,

@Warenlagereingaenge XML,

-- Optionen für die Auslieferung (Bitmaske)
-- 0x0001 = Pro Bestellung eine Pickliste anlegen
-- 0x0002 = Chargen automatisch vorbelegen
@nOptions INT,

-- kBenutzer
@kBenutzer INT,

-- Sitzungs-Id
@kSessionId INT

AS  
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

DECLARE @TranCounter INT;
SET @TranCounter = @@TRANCOUNT;
IF @TranCounter > 0
	SAVE TRANSACTION spReserviereBestellungen
ELSE
	BEGIN TRANSACTION;

BEGIN TRY
	DECLARE @kBestellung INT;	
	DECLARE @kBestellPos BIGINT;
	DECLARE @fAnzahl DECIMAL(28,14);

	DECLARE cBestellungen CURSOR LOCAL FAST_FORWARD FOR
		SELECT ParamValues.ID.value('kBestellung[1]', 'INT') AS kBestellung
		FROM @Bestellungen.nodes('/Bestellung') AS ParamValues(ID)
		ORDER BY ISNULL(ParamValues.ID.value('nPrio[1]', 'INT'), 0) ASC;
	OPEN cBestellungen;
	FETCH NEXT FROM cBestellungen INTO @kBestellung;
	WHILE @@FETCH_STATUS = 0
	BEGIN
		DECLARE cBestellpositionen CURSOR LOCAL FAST_FORWARD FOR
			SELECT 
				Auslieferung.vBestellPos.kBestellPos AS kBestellpos, 
				Auslieferung.vBestellPos.fAnzahlOffen AS fAnzahl
			FROM 
				Auslieferung.vBestellPos
			WHERE Auslieferung.vBestellPos.kBestellung = @kBestellung AND 
				Auslieferung.vBestellPos.fAnzahlOffen > 0.0 AND -- Menge größer 0 
				(((@nOptions & 2) > 0) OR Auslieferung.vBestellPos.nIstChargenartikel = 0) AND -- Chargenartikel nur wenn gewünscht
                Auslieferung.vBestellPos.kBestellStueckliste != Auslieferung.vBestellPos.kBestellPos -- keine Stücklistenväter

		OPEN cBestellpositionen;
		FETCH NEXT FROM cBestellpositionen INTO @kBestellpos, @fAnzahl;
		WHILE @@FETCH_STATUS = 0
		BEGIN
			EXEC Auslieferung.spReserviereBestellposition
				@kBestellPos = @kBestellPos,
				@fAnzahl = @fAnzahl,
				@nOptions = @nOptions,
				@Laeger = @Laeger,
				@Warenlagereingaenge = @Warenlagereingaenge,
				@kBenutzer = @kBenutzer,
				@kSessionId = @kSessionId;

			FETCH NEXT FROM cBestellpositionen INTO @kBestellpos, @fAnzahl;
		END
		CLOSE cBestellpositionen;
		DEALLOCATE cBestellpositionen;

		FETCH NEXT FROM cBestellungen INTO @kBestellung;
	END
	CLOSE cBestellungen;
	DEALLOCATE cBestellungen;

	EXEC [Auslieferung].[spPicklistenStuecklistenFix]
		  @kBenutzer = @kBenutzer,
		  @kSessionId = @kSessionId

	IF @TranCounter = 0
		COMMIT TRANSACTION;
END TRY
BEGIN CATCH
	IF @TranCounter = 0
		ROLLBACK TRANSACTION;
	ELSE
		IF XACT_STATE() <> -1
			ROLLBACK TRANSACTION spReserviereBestellungen;

	DECLARE @ErrorMessage NVARCHAR(4000);
    DECLARE @ErrorSeverity INT;
    DECLARE @ErrorState INT;

    SELECT @ErrorMessage = ERROR_MESSAGE();
    SELECT @ErrorSeverity = ERROR_SEVERITY();
    SELECT @ErrorState = ERROR_STATE();

    RAISERROR (@ErrorMessage, -- Message text.
               @ErrorSeverity, -- Severity.
               @ErrorState -- State.
              );
END CATCH
