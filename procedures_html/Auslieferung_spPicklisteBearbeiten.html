CREATE PROCEDURE [Auslieferung].[spPicklisteBearbeiten]

-- Ändert den Status einer Pickliste für die Bearbeitung in einer Sitzung
--
-- Transaction-Handling: siehe https://msdn.microsoft.com/en-us/library/ms188378.aspx
--

@kPickliste INT,

-- Enthält den Benutzer zur eindeutigen Zuordnung
@kBenutzer INT,

@kSessionId INT

-- Mögliche Fehler:
-- Im Fehlerfall enthält der Messagetext XML-Informationen zum Fehler
-- Code = 50000, Severity = 18, State = 4: Die Pickliste wird bereits bearbeitet
-- Code = 50000, Severity = 18, State = 5: Die Pickliste ist bereits ausgeliefert

AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

DECLARE @TranCounter INT;
SET @TranCounter = @@TRANCOUNT;
IF @TranCounter > 0
	SAVE TRANSACTION spPicklisteBearbeiten
ELSE
	BEGIN TRANSACTION;

BEGIN TRY
	DECLARE @xErrorInfo XML;
	DECLARE @cError NVARCHAR(MAX);

	-- Status prüfen
	SET @xErrorInfo = (SELECT
			dbo.tPickliste.kPickliste,
			dbo.tPickliste.nStatus,
			dbo.tSessionId.kBenutzer,
			dbo.tSessionId.cRechnername,
			dbo.tSessionId.dLastAction
		FROM dbo.tPickliste
		LEFT JOIN dbo.tSessionId ON dbo.tPickliste.kSessionId = dbo.tSessionId.kSessionId

		WHERE dbo.tPickliste.nStatus < 10 AND dbo.tPickliste.kPickliste = @kPickliste AND dbo.tPickliste.kSessionId <> @kSessionId
		FOR XML PATH('PicklisteInBearbeitung'), TYPE)
	IF (@xErrorInfo IS NOT NULL)
	BEGIN			
		SET @cError = CAST(@xErrorInfo AS NVARCHAR(MAX))

		-- 09.07.2018	MAD
		-- Richtiges Rollback für die Transaktion durchführen und nicht für die gesamte Transaktion
		IF @TranCounter = 0
			ROLLBACK TRANSACTION;
		ELSE
			IF XACT_STATE() <> -1
				ROLLBACK TRANSACTION spPicklisteBearbeiten;

		RAISERROR(@cError, 18, 4);
		RETURN;
	END

	SET @xErrorInfo = (SELECT			
			dbo.tPickliste.kPickliste,
			dbo.tPickliste.nStatus,
			dbo.tWMSVersandLock.kBenutzer,
			'Unbekannter Rechner (Packtisch/WMS)' AS cRechnername,
			dbo.tWMSVersandLock.dDatum AS dLastAction
		FROM dbo.tWMSVersandLock
		JOIN dbo.tPickliste ON dbo.tPickliste.kPickliste = tWMSVersandLock.kPickliste
		WHERE dbo.tWMSVersandLock.kPickliste = @kPickliste
		FOR XML PATH('PicklisteInBearbeitung'), TYPE)
	IF (@xErrorInfo IS NOT NULL)
	BEGIN			
		SET @cError = CAST(@xErrorInfo AS NVARCHAR(MAX))

		-- 09.07.2018	MAD
		-- Richtiges Rollback für die Transaktion durchführen und nicht für die gesamte Transaktion
		IF @TranCounter = 0
			ROLLBACK TRANSACTION;
		ELSE
			IF XACT_STATE() <> -1
				ROLLBACK TRANSACTION spPicklisteBearbeiten;

		RAISERROR(@cError, 18, 4);
		RETURN;
	END

	-- Status prüfen
	SET @xErrorInfo = (SELECT
			dbo.tPickliste.kPickliste,
			dbo.tPickliste.nStatus,
			dbo.tSessionId.kBenutzer,
			dbo.tSessionId.cRechnername,
			dbo.tSessionId.dLastAction
		FROM dbo.tPickliste
		LEFT JOIN dbo.tSessionId ON dbo.tPickliste.kSessionId = dbo.tSessionId.kSessionId
		WHERE dbo.tPickliste.nStatus > 11 AND dbo.tPickliste.kPickliste = @kPickliste
		FOR XML PATH('PicklisteAusgeliefert'), TYPE)
	IF (@xErrorInfo IS NOT NULL)
	BEGIN			
		SET @cError = CAST(@xErrorInfo AS NVARCHAR(MAX))
	
		-- 09.07.2018	MAD
		-- Richtiges Rollback für die Transaktion durchführen und nicht für die gesamte Transaktion
		IF @TranCounter = 0
			ROLLBACK TRANSACTION;
		ELSE
			IF XACT_STATE() <> -1
				ROLLBACK TRANSACTION spPicklisteBearbeiten;
	
		RAISERROR(@cError, 18, 5);
		RETURN;
	END

	-- Session in Pickliste setzen, sofern die Pickliste noch nicht bearbeitet wird
	UPDATE dbo.tPickliste
	SET
	    dbo.tPickliste.kSessionId = @kSessionId
	WHERE
		dbo.tPickliste.kPickliste = @kPickliste

	-- Status der Pickliste ändern
	INSERT INTO dbo.tPicklistePosStatus
	(
	    dbo.tPicklistePosStatus.kPicklistePos,
	    dbo.tPicklistePosStatus.kbenutzer,
	    dbo.tPicklistePosStatus.dZeitstempel,
	    dbo.tPicklistePosStatus.nStatus
	)
	SELECT 
		dbo.tPicklistePos.kPicklistePos,
		@kBenutzer,
		GETDATE(),
		5
	FROM dbo.tPicklistePos 
	JOIN dbo.tPickliste ON dbo.tPickliste.kPickliste = dbo.tPicklistePos.kPickliste
	LEFT JOIN Verkauf.tAuftragPosition ON tAuftragPosition.kAuftragPosition = tPicklistePos.kBestellPos
	WHERE dbo.tPickliste.kPickliste = @kPickliste 
		AND dbo.tPickliste.kSessionId = @kSessionId 
		AND (tPicklistePos.nStatus <= 10 OR (tAuftragPosition.nType IN(18, 20) AND tPicklistePos.nStatus <= 20))
    
	IF @TranCounter = 0
		COMMIT TRANSACTION;
END TRY
BEGIN CATCH
	IF @TranCounter = 0
		ROLLBACK TRANSACTION;
	ELSE
		IF XACT_STATE() <> -1
			ROLLBACK TRANSACTION spPicklisteBearbeiten;

	DECLARE @ErrorMessage NVARCHAR(4000);
    DECLARE @ErrorSeverity INT;
    DECLARE @ErrorState INT;

    SELECT @ErrorMessage = ERROR_MESSAGE();
    SELECT @ErrorSeverity = ERROR_SEVERITY();
    SELECT @ErrorState = ERROR_STATE();

    RAISERROR (@ErrorMessage, -- Message text.
               @ErrorSeverity, -- Severity.
               @ErrorState -- State.
              );
END CATCH
