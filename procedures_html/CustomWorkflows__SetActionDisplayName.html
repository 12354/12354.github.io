CREATE PROCEDURE [CustomWorkflows].[_SetActionDisplayName]
    @actionName VARCHAR(128),
    @displayName VARCHAR(128)
AS
BEGIN

    IF EXISTS
    (
        SELECT Prods.name AS cName,
               ISNULL(displayName.value, Prods.name) AS cDisplayName
        FROM sys.procedures AS Prods
            JOIN sys.schemas Schem
                ON Schem.schema_id = Prods.schema_id
            JOIN sys.extended_properties displayName
                ON displayName.major_id = Prods.object_id
                   AND displayName.name = 'DisplayName'
        WHERE Schem.name = 'CustomWorkflows'
              AND Prods.name = @actionName
    )
    BEGIN

        EXEC sys.sp_dropextendedproperty @name = N'DisplayName',
                                         @level0type = N'SCHEMA',
                                         @level0name = N'CustomWorkflows',
                                         @level1type = N'PROCEDURE',
                                         @level1name = @actionName;
    END;

    EXEC sys.sp_addextendedproperty @name = N'DisplayName',
                                    @value = @displayName,
                                    @level0type = 'SCHEMA',
                                    @level0name = N'CustomWorkflows',
                                    @level1type = 'PROCEDURE',
                                    @level1name = @actionName,
                                    @level2type = NULL,
                                    @level2name = NULL;

END;
