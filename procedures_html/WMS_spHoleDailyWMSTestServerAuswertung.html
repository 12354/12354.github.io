CREATE PROCEDURE [WMS].[spHoleDailyWMSTestServerAuswertung]
@nHoleAlles TINYINT = 0

--
-- Diese SP liefert Statistiken zu WMS Performancetests
-- !!! DIESE SP WIRD DARF NICHT PRODUKTIV VERDENDET !!!!
--


AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

	

BEGIN 

	IF(@nHoleAlles != 0) 
	BEGIN

		SELECT 'Picklisten Statistiken'

		SELECT * FROM WMS.tStatistikPicklisten
		WHERE dTimeStamp >=DATEADD(day, DATEDIFF(day,0,GETDATE()),0) 
		AND dTimeStamp < DATEADD(day, DATEDIFF(day,0,GETDATE())+1,0);

		SELECT '----------------------'

		SELECT 'Eazyshipping Statistiken';

		SELECT * FROM WMS.tStatistikEazyShipping
		WHERE dTimeStamp >=DATEADD(day, DATEDIFF(day,0,GETDATE()),0) 
		AND dTimeStamp < DATEADD(day, DATEDIFF(day,0,GETDATE())+1,0);
	END
	ELSE
	BEGIN

		SELECT 'Picklisten Statistiken';

		SELECT * FROM WMS.tStatistikPicklisten
		ORDER BY dTimeStamp;

		SELECT '----------------------';

		SELECT 'Eazyshipping Statistiken';

		SELECT * FROM WMS.tStatistikEazyShipping
		ORDER BY dTimeStamp;

	END;


END;
