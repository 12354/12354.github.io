CREATE PROCEDURE FulfillmentNetwork.[spAbrechnungEinzelnachweisLoeschen]
    @kAbrechnungEinzelnachweis INT
AS
BEGIN
    DECLARE @kAbrechnungPos AS INT;
    DECLARE @fAnzahl AS DECIMAL(25,13);
    DECLARE @fAnzahlNeu AS DECIMAL(25,13);
    SELECT  @kAbrechnungPos = Einzelnachweis.kAbrechnungPos ,
            @fAnzahl = Einzelnachweis.fAnzahl
    FROM    FulfillmentNetwork.tAbrechnungEinzelnachweis AS Einzelnachweis
    WHERE   Einzelnachweis.kAbrechnungEinzelnachweis = @kAbrechnungEinzelnachweis;

			
    SELECT  @fAnzahlNeu = fAnzahl - @fAnzahl
    FROM    FulfillmentNetwork.tAbrechnungPos
    WHERE   kAbrechnungPos = @kAbrechnungPos;
    IF ( @fAnzahlNeu > 0 )
        BEGIN
            UPDATE  FulfillmentNetwork.tAbrechnungPos
            SET     fAnzahl = @fAnzahlNeu
            WHERE   kAbrechnungPos = @kAbrechnungPos;

        END;
    ELSE
        BEGIN
            DELETE  FROM FulfillmentNetwork.tAbrechnungPos
            WHERE   kAbrechnungPos = @kAbrechnungPos;
        END;

	
    DELETE  FROM FulfillmentNetwork.tAbrechnungEinzelnachweis
    WHERE   kAbrechnungEinzelnachweis = @kAbrechnungEinzelnachweis;
END
