CREATE PROCEDURE [Maintenance].[spGarbageCollection_VerwaistePicklistenLoeschen]
AS
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

BEGIN
	DECLARE @Picklisten TABLE
	(
		kPickliste INT NOT NULL
	)

	INSERT @Picklisten (kPickliste)
		SELECT
			kPickliste
		FROM dbo.tPickliste
		LEFT JOIN dbo.tSessionId ON tSessionId.kSessionId = tPickliste.kSessionId
		WHERE (tSessionId.kSessionId IS NULL OR DATEADD(mi, 10, tSessionId.dLastAction) < GETDATE()) 
			AND (tPickliste.nStatus = 5 OR tPickliste.nStatus = 10 AND (tPickliste.kLieferant > 0 OR tPickliste.kAnsprechpartner > 0));

	DELETE dbo.tPickliste
	FROM dbo.tPickliste
	JOIN @Picklisten AS PicklistenTable ON tPickliste.kPickliste = PicklistenTable.kPickliste;
END
