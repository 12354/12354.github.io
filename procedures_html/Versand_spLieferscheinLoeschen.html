Versand.spLieferscheinLoeschen

Description: Mit der Stored Procedure können Einträge in der Tabelle dbo.tLieferschein gelöscht werden.

Long Description: Keine ausführliche Beschreibung hinterlegt.

Created At: 2024-01-12 03:32:04

Updated At: 2024-01-12 03:32:04

Code:

CREATE PROCEDURE [Versand].[spLieferscheinLoeschen]
	@xLieferschein XML = NULL,
	@kLieferschein INT = NULL
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @retry INT;
DECLARE @OldContextInfo varbinary(128);
DECLARE @CreatedTransaction INT;
IF(CONTEXT_INFO() IS NOT NULL)
	SET @OldContextInfo = CONTEXT_INFO();
ELSE
	SET @OldContextInfo = 0x0;
SET @retry = 5;
WHILE @retry > 0
BEGIN
	SET CONTEXT_INFO 0x5114;
	BEGIN TRY
		IF (@@TRANCOUNT = 0)
		BEGIN
			SET @CreatedTransaction = 1;
			BEGIN TRANSACTION
		END
		ELSE
		BEGIN
			SET @CreatedTransaction = 0;
			SAVE TRANSACTION SavePointLieferscheinLoeschen;
		END
			IF(OBJECT_ID('tempdb..#Lieferschein') IS NOT NULL)
			BEGIN
				DROP TABLE #Lieferschein;
			END
			CREATE TABLE #Lieferschein(kLieferschein INT, kBestellung INT);
			IF(@xLieferschein IS NOT NULL)
			BEGIN
				INSERT INTO #Lieferschein(kLieferschein, kBestellung)
					SELECT 
						Lieferschein.ID.value('kLieferschein[1]' , 'INT'),
						dbo.tlieferschein.kBestellung
						FROM @xLieferschein.nodes('Lieferschein') AS Lieferschein(ID)
						JOIN dbo.tlieferschein ON dbo.tlieferschein.kLieferschein = Lieferschein.ID.value('kLieferschein[1]' , 'INT')
			END
			ELSE
			BEGIN
				INSERT INTO #Lieferschein(kLieferschein, kBestellung)
				SELECT dbo.tlieferschein.kLieferschein, dbo.tlieferschein.kBestellung
				FROM dbo.tlieferschein WHERE dbo.tlieferschein.kLieferschein = @kLieferschein
			END
			
			DELETE FROM dbo.tVersand
				WHERE dbo.tVersand.kLieferschein IN (SELECT kLieferschein FROM #Lieferschein);

			DECLARE @xLieferscheinPos TYPE_spLieferscheinPosLoeschen;

			INSERT INTO @xLieferscheinPos
			SELECT dbo.tlieferscheinPos.kLieferscheinPos 
			FROM #Lieferschein 
			JOIN dbo.tlieferscheinPos ON #Lieferschein.kLieferschein = dbo.tlieferscheinPos.kLieferschein
					
			EXEC Versand.spLieferscheinPosLoeschen @xLieferscheinPos;

			-- Bestellung zum Shop senden da sich nKomplettAusgeliefert geändert hat
			UPDATE Verkauf.tAuftrag
				SET tAuftrag.cInet = 'Y'
				WHERE tAuftrag.kShopauftrag > 0 AND tAuftrag.kAuftrag IN (SELECT kBestellung FROM #Lieferschein);

			-- Alle Einträge aus tQueue für die betroffenen Lieferscheine entfernen - Löschen gewinnt
			DELETE dbo.tQueue
			FROM dbo.tQueue
				WHERE dbo.tQueue.cName = 'tLieferschein' AND dbo.tQueue.kWert IN (SELECT kLieferschein FROM #Lieferschein);

			-- Löscheinträge in tQueue schreiben
			INSERT INTO tQueue (kShop, kPlattform, cName, kWert, nAction)
			SELECT
				vAuftrag.kShop,
				0, -- kPlattform
				'tLieferschein',
				#Lieferschein.kLieferschein,
				2
			FROM #Lieferschein
			JOIN Verkauf.vAuftrag ON vAuftrag.kAuftrag = #Lieferschein.kBestellung
			WHERE vAuftrag.kShopauftrag > 0 AND vAuftrag.kShop > 0

			DELETE dbo.tlieferschein
				WHERE dbo.tlieferschein.kLieferschein IN (SELECT kLieferschein FROM #Lieferschein);
			IF(@CreatedTransaction = 1)
			BEGIN				
				COMMIT
			END
			SET @retry = -1;
			SET CONTEXT_INFO @OldContextInfo;		
	END TRY
	BEGIN CATCH
	IF(ERROR_NUMBER() = 1205)
			BEGIN
				SET @retry = @retry - 1;
				IF(@CreatedTransaction = 1)
				BEGIN
					ROLLBACK TRANSACTION;
				END
				ELSE
				BEGIN
					ROLLBACK TRANSACTION SavePointLieferscheinLoeschen;
				END
				IF(@retry = 0)
				BEGIN
					SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
					RAISERROR (	@ErrorMessage, 
								@ErrorSeverity,
								@ErrorState
					);
					SET CONTEXT_INFO 0x0;
					RETURN;
				END
				EXEC dbo.spWaitRandom;				
			END
			ELSE 
			BEGIN
				SET @retry = -1;
				SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
				IF(@CreatedTransaction = 1)
				BEGIN
					ROLLBACK TRANSACTION;
				END
				ELSE
				BEGIN
					ROLLBACK TRANSACTION SavePointLieferscheinLoeschen;
				END
				RAISERROR (	@ErrorMessage, 
							@ErrorSeverity,
							@ErrorState
				);
				SET CONTEXT_INFO @OldContextInfo;
				RETURN;
			END
	END CATCH
	SET CONTEXT_INFO @OldContextInfo;	
END
