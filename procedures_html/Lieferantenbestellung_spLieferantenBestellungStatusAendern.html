CREATE PROCEDURE [Lieferantenbestellung].[spLieferantenBestellungStatusAendern] 
	@kLieferantenBestellung INT,
	@nStatus INT
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
BEGIN TRANSACTION
		DECLARE @kLieferant INT;
		DECLARE @kSprache INT;
		DECLARE @kLieferantenBestellungRA INT;
		DECLARE @kLieferantenBestellungLA INT;
		DECLARE @cWaehrungISO NVARCHAR(5);
		DECLARE @cInternerKommentar NVARCHAR(MAX);
		DECLARE @cDruckAnmerkung NVARCHAR(MAX);
		DECLARE @dGedruckt DATETIME;
		DECLARE @dGemailt DATETIME;
		DECLARE @dGefaxt DATETIME;
		DECLARE @dErstellt DATETIME;
		DECLARE @kFirma INT;
		DECLARE @kLager INT;
		DECLARE @kKunde INT;
		DECLARE @dLieferdatum DATETIME;
		DECLARE @cEigeneBestellnummer NVARCHAR(255);
		DECLARE @cBezugsAuftragsNummer NVARCHAR(255);
		DECLARE @nDropShipping INT;
		DECLARE @kLieferantenBestellungLieferant INT;
		DECLARE @kBenutzer INT;
		DECLARE @fFaktor DECIMAL(25,13);
		DECLARE @dAngemahnt DATETIME;
		DECLARE @dInBearbeitung DATETIME;
		DECLARE @nDeleted INT;
		DECLARE @nManuellAbgeschlossen TINYINT;
		DECLARE @cFremdbelegnummer NVARCHAR(255);
		DECLARE @kLieferschein INT;
		DECLARE @nBestaetigt TINYINT;
		DECLARE @dExportiert DATETIME;
		SELECT	@kLieferant = dbo.tLieferantenBestellung.kLieferant,
				@kSprache = dbo.tLieferantenBestellung.kSprache,
				@kLieferantenBestellungRA = dbo.tLieferantenBestellung.kLieferantenBestellungRA,
				@kLieferantenBestellungLA = dbo.tLieferantenBestellung.kLieferantenBestellungLA,
				@cWaehrungISO = dbo.tLieferantenBestellung.cWaehrungISO,
				@cInternerKommentar = dbo.tLieferantenBestellung.cInternerKommentar,
				@cDruckAnmerkung = dbo.tLieferantenBestellung.cDruckAnmerkung,
				@dGedruckt = dbo.tLieferantenBestellung.dGedruckt,
				@dGemailt = dbo.tLieferantenBestellung.dGemailt,
				@dGefaxt = dbo.tLieferantenBestellung.dGefaxt,
				@dErstellt = dbo.tLieferantenBestellung.dErstellt,
				@kFirma = dbo.tLieferantenBestellung.kFirma,
				@kLager = dbo.tLieferantenBestellung.kLager,
				@kKunde = dbo.tLieferantenBestellung.kKunde,
				@dLieferdatum = dbo.tLieferantenBestellung.dLieferdatum,
				@cEigeneBestellnummer = dbo.tLieferantenBestellung.cEigeneBestellnummer,
				@cBezugsAuftragsNummer = dbo.tLieferantenBestellung.cBezugsAuftragsNummer,
				@nDropShipping = dbo.tLieferantenBestellung.nDropShipping,
				@kLieferantenBestellungLieferant = dbo.tLieferantenBestellung.kLieferantenBestellungLieferant,
				@kBenutzer = dbo.tLieferantenBestellung.kBenutzer,
				@fFaktor = dbo.tLieferantenBestellung.fFaktor,
				@dAngemahnt = dbo.tLieferantenBestellung.dAngemahnt,
				@dInBearbeitung = dbo.tLieferantenBestellung.dInBearbeitung,
				@nDeleted = dbo.tLieferantenBestellung.nDeleted,
				@nManuellAbgeschlossen = dbo.tLieferantenBestellung.nManuellAbgeschlossen,
				@cFremdbelegnummer = dbo.tLieferantenBestellung.cFremdbelegnummer,
				@kLieferschein = dbo.tLieferantenBestellung.kLieferschein, 
				@nBestaetigt = dbo.tLieferantenBestellung.nBestaetigt,
				@dExportiert = dbo.tLieferantenBestellung.dExportiert 
			FROM dbo.tLieferantenBestellung
			WHERE dbo.tLieferantenBestellung.kLieferantenBestellung = @kLieferantenBestellung;
		EXEC Lieferantenbestellung.spLieferantenBestellungBearbeiten @kLieferantenBestellung, @kLieferant, @kSprache, @kLieferantenBestellungRA, @kLieferantenBestellungLA, @cWaehrungISO,
				@cInternerKommentar, @cDruckAnmerkung, @dGedruckt, @dGemailt, @dGefaxt, @nStatus, @dErstellt, @kFirma, @kLager, @kKunde, @dLieferdatum,
				@cEigeneBestellnummer, @cBezugsAuftragsNummer, @nDropShipping, @kLieferantenBestellungLieferant, @kBenutzer, @fFaktor, @dAngemahnt,
				@dInBearbeitung, @nDeleted, @nManuellAbgeschlossen, @cFremdbelegnummer, @kLieferschein,@nBestaetigt, @dExportiert;
COMMIT
