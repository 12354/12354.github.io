CREATE PROCEDURE [ArtikelVerwaltung].[spDeleteArtikelFromKategorie]
	@artikelKeys [dbo].[TYPE_spDeleteArtikelFromKategorie] READONLY,
	@targetKategorie INT
AS
BEGIN
	DELETE 
	FROM dbo.tKategorieartikel
    WHERE tKategorieartikel.kKategorieArtikel IN 
	(
		SELECT tkategorieartikel.kKategorieArtikel
		FROM dbo.tkategorieartikel
        JOIN dbo.tArtikel 
			ON tKategorieartikel.kArtikel = tArtikel.kArtikel 
        WHERE 
		(
			tArtikel.kArtikel IN 
			(
				SELECT kArtikel 
				FROM @artikelKeys
			) 
			OR tArtikel.kVaterArtikel IN 
			(
				SELECT kArtikel
				FROM @artikelKeys
			)
		) 
		AND (tkategorieartikel.kKategorie = @targetKategorie)
	)
END
