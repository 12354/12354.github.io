CREATE PROCEDURE [Maintenance].[spBestandAktualisieren]
	@kArtikel INT = NULL,
	@cArtNr NVARCHAR(100) = NULL
AS
BEGIN
	IF(OBJECT_ID('tempdb..#FehlendeEintraege') IS NOT NULL)
	BEGIN
		DROP TABLE #FehlendeEintraege;
	END
	CREATE TABLE #FehlendeEintraege(kArtikel INT, kWarenlager INT);
	DECLARE @lagerbestandProLager AS dbo.TYPE_spUpdateLagerbestandProLager;
	DECLARE @lagerbestand AS dbo.TYPE_spUpdateLagerbestand;
	IF(@kArtikel IS NOT NULL)
	BEGIN
		INSERT INTO @lagerbestandProLager(kArtikel, kWarenlager)
			SELECT @kArtikel, kWarenLager
				FROM dbo.tWarenLager;
		INSERT INTO @lagerbestand(kArtikel)
			VALUES (@kArtikel);
	END
	ELSE IF(@cArtNr IS NOT NULL)
	BEGIN
		INSERT INTO @lagerbestandProLager(kArtikel, kWarenlager)
			SELECT tArtikel.kArtikel, kWarenlager
				FROM dbo.tArtikel
				CROSS JOIN dbo.tWarenLager
				WHERE tArtikel.cArtNr = @cArtNr;
		INSERT INTO @lagerbestand(kArtikel)
			SELECT tArtikel.kArtikel
				FROM dbo.tArtikel
				WHERE tArtikel.cArtNr = @cArtNr;
	END
	ELSE
	BEGIN
		INSERT INTO @lagerbestandProLager(kArtikel, kWarenlager)
			SELECT kArtikel, kWarenlager
				FROM Maintenance.vLagerbestandProLager;
		INSERT INTO @lagerbestand(kArtikel)
			SELECT kArtikel
				FROM Maintenance.vBestand;
	END
	IF(EXISTS(SELECT * FROM @lagerbestandProLager))
	BEGIN
		INSERT INTO dbo.tlagerbestandProLagerLagerartikel(kArtikel, kWarenlager, fBestand)
			SELECT [@lagerbestandProLager].kArtikel, [@lagerbestandProLager].kWarenlager, 0.0
				FROM @lagerbestandProLager
				LEFT JOIN dbo.tlagerbestandProLagerLagerartikel ON tlagerbestandProLagerLagerartikel.kArtikel = [@lagerbestandProLager].kArtikel
																AND tlagerbestandProLagerLagerartikel.kWarenlager = [@lagerbestandProLager].kWarenlager
				WHERE tlagerbestandProLagerLagerartikel.kArtikel IS NULL;
		EXEC dbo.spUpdateLagerbestandProLager @Artikel = @lagerbestandProLager;
	END
	IF(EXISTS(SELECT * FROM @lagerbestand))
	BEGIN
		EXEC dbo.spUpdateLagerbestand @Artikel = @lagerbestand;
	END

END
