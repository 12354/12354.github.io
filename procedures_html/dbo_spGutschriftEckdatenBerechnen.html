CREATE PROCEDURE [dbo].[spGutschriftEckdatenBerechnen]
	@Gutschriften AS dbo.TYPE_spGutschriftEckdatenBerechnen READONLY
AS
BEGIN
	IF NOT EXISTS(SELECT 1 FROM @Gutschriften)
	BEGIN
		RETURN 0;
	END

	DECLARE
		@OldContextInfo VARBINARY(128),
		@Hash VARBINARY(128);

	SET @OldContextInfo = ISNULL(CONTEXT_INFO(), 0x0);
	SET @Hash = HASHBYTES('SHA1', 'dbo.spGutschriftEckdatenBerechnen');
	SET CONTEXT_INFO @Hash;

	-- ---------- Gutschriftseckdatenberechnen und Gutschrift als valide kennzeichnen ----------
	UPDATE dbo.tgutschrift
		SET
			fPreis = PosDaten.BruttoPreis,
			nGutschriftStatus = 0
		FROM dbo.tgutschrift
		JOIN @Gutschriften AS Gutschriften ON Gutschriften.kGutschrift = tgutschrift.kGutschrift
		JOIN (
			SELECT 
				tGutschriftPos.tGutschrift_kGutschrift,
				ROUND(SUM(tGutschriftPos.nAnzahl * tGutschriftPos.fVKNetto * (1 - (tGutschriftPos.fRabatt / 100)) * (1 + (tGutschriftPos.fMwSt / 100))), 2) AS BruttoPreis
			FROM dbo.tGutschriftPos
			GROUP BY tGutschriftPos.tGutschrift_kGutschrift
		) AS PosDaten ON PosDaten.tGutschrift_kGutschrift = tgutschrift.kGutschrift;

	-- ---------- Rechnungseckdaten berechnen ----------
	DECLARE @tempRechnungsKeys AS Rechnung.TYPE_spRechnungEckdatenBerechnen

	INSERT INTO @tempRechnungsKeys (kRechnung)
	SELECT		DISTINCT tgutschrift.kRechnung
	FROM		dbo.tgutschrift
	INNER JOIN	@Gutschriften AS Gutschriften ON Gutschriften.kGutschrift = tgutschrift.kGutschrift;

	EXEC Rechnung.spRechnungEckdatenBerechnen @tempRechnungsKeys;

	SET CONTEXT_INFO @OldContextInfo;
END
