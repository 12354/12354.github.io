CREATE PROCEDURE SCX.spLoescheKaputteExterneRechnungskorrektur
	@kGutschrift INT,
	@kBenutzer INT
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT ON;
BEGIN TRY
	DECLARE @DeadlockRetries TINYINT = CASE WHEN @@TRANCOUNT = 0 AND XACT_STATE() = 0 THEN 5 ELSE 0 END;

	IF ISNULL(@kGutschrift, 0) = 0 OR ISNULL(@kBenutzer, 0) = 0
	BEGIN
		RETURN;
	END

	IF NOT EXISTS (SELECT 1 FROM dbo.tgutschrift
				JOIN Rechnung.tRechnung ON tRechnung.kRechnung = tgutschrift.kRechnung
				JOIN dbo.tShop ON tShop.kShop = tRechnung.kShop
				WHERE tgutschrift.kGutschrift = @kGutschrift
					AND tShop.nTyp = 6
					AND tRechnung.nIstExterneRechnung > 0)
	BEGIN
		RETURN;
	END

	DeadlockRetry:
    BEGIN TRY
        IF (@DeadlockRetries > 0) BEGIN TRANSACTION;

		DECLARE @xPositionen AS XML;
		SET @xPositionen = 
		(
			SELECT tGutschriftPos.kGutschriftPos
			FROM dbo.tgutschrift
			JOIN dbo.tGutschriftPos ON tgutschrift.kGutschrift = tGutschriftPos.tGutschrift_kGutschrift
			WHERE tgutschrift.kGutschrift = @kGutschrift

			FOR XML PATH('Gutschriftpos'), TYPE
		);

		EXEC Gutschrift.spGutschriftposLoeschen @xPositionen;

        DELETE FROM dbo.tZahlung
        WHERE tZahlung.kGutschrift = @kGutschrift;
        
        DELETE FROM dbo.tgutschrift
        WHERE tgutschrift.kGutschrift = @kGutschrift;

		-- Ende vom Code
        IF (@DeadlockRetries > 0)
           AND (@@TRANCOUNT > 0)
           AND (XACT_STATE() = 1) COMMIT TRANSACTION;
	END TRY
    BEGIN CATCH
        BEGIN TRY
            EXEC dbo.spErrorHandler @DeadlockRetries = @DeadlockRetries OUTPUT;
            IF (@DeadlockRetries > 0) GOTO DeadlockRetry;
        END TRY
        BEGIN CATCH
            IF ERROR_NUMBER() <> 266 THROW;
        END CATCH;
        THROW;
    END CATCH;
END TRY
BEGIN CATCH
    THROW;
END CATCH;
