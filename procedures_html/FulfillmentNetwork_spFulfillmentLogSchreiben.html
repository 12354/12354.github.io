CREATE PROCEDURE FulfillmentNetwork.spFulfillmentLogSchreiben
    @FulfillemntLogDaten AS FulfillmentNetwork.[TYPE_spFulfillmentLogSchreiben] READONLY,
    @kVorgang INT
AS
BEGIN


	declare @IdentityOutput table ( ID int )
    INSERT INTO FulfillmentNetwork.tFulfillmentLog
    (
        kBenutzer,
        dTimestamp,
        cSessionId,
        nLogLevel,
        nMessageSource,
        kLieferant,
        kLieferschein,
        kArtikelHistory,
        kWarenlager,
        kBestellung,
        kLieferantenBestellung,
        kFulfillmentAuftrag,
        kArtikel,
        cMessage,
        cRequestId,
        kKunde,
        cMessageDetails
    )
	OUTPUT Inserted.IDENTITYCOL INTO @IdentityOutput
    SELECT kBenutzer,
           dTimestamp,
           cSessionId,
           nLogLevel,
           nMessageSource,
           kLieferant,
           kLieferschein,
           kArtikelHistory,
           kWarenlager,
           kBestellung,
           kLieferantenBestellung,
           kFulfillmentAuftrag,
           kArtikel,
           cMessage,
           cRequestId,
           kKunde,
           cMessageDetails FROM @FulfillemntLogDaten 	
	
	IF(@kVorgang > 0)
	BEGIN
		INSERT INTO FulfillmentNetwork.tVorgang_FulfillmentLog	
		SELECT @kVorgang,ID FROM @IdentityOutput 
	END
END;
