CREATE PROCEDURE [FulfillmentNetwork].[spProductUpdate] 
	@Daten AS FulfillmentNetwork.TYPE_spProductUpdate READONLY,
	@kBenutzer AS INT
AS 
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @OldContextInfo VARBINARY(128);
DECLARE @retry INT;
DECLARE @CreatedTransaction INT;
DECLARE @ReturnValue INT;
SET @retry = 5;
WHILE @retry > 0
BEGIN
	-- Context Info
	IF(CONTEXT_INFO() IS NOT NULL)
	BEGIN
		SET @OldContextInfo = CONTEXT_INFO();
	END
	ELSE
	BEGIN
		SET @OldContextInfo = 0x0;
	END
	SET CONTEXT_INFO 0x696969 -- ContextInfo festlegen
	BEGIN TRY
		IF (@@TRANCOUNT = 0)
		BEGIN
		   SET @CreatedTransaction = 1;
		   BEGIN TRANSACTION
		END;
		ELSE
		BEGIN
			SET @CreatedTransaction = 0;
			SAVE TRANSACTION Savepoint0;
		END;
			IF(OBJECT_ID('tempdb..#Daten_spProductUpdate') IS NOT NULL)
		BEGIN
			DROP TABLE #Daten_spProductUpdate;
		END
		CREATE TABLE #Daten_spProductUpdate(kArtikel INT, cJfsku Nvarchar(11));
		INSERT INTO #Daten_spProductUpdate(kArtikel, cJfsku)
			SELECT Daten.kArtikel, 
				CASE WHEN Daten.Jfsku IS NOT NULL OR Daten.xFlag_Jfsku = 1
					THEN Daten.Jfsku
					ELSE tArtikel.cJfsku
				END
			
			FROM @Daten AS Daten
				JOIN dbo.tArtikel ON Daten.kArtikel = tArtikel.kArtikel
	--
	-- Hier kommt der Code rein vor dem INSERT/UPDATE/DELETE
	--
	IF(OBJECT_ID('tempdb..#Temp_spProductUpdate') IS NOT NULL)
	BEGIN
		DROP TABLE #Temp_spProductUpdate;
	END
	CREATE TABLE #Temp_spProductUpdate(InternalId INT IDENTITY(1,1), kKunde INT);
		-- Update schreiben
		UPDATE dbo.tArtikel			
			SET cJfsku = Daten.cJfsku
			OUTPUT INSERTED.kArtikel INTO #Temp_spProductUpdate 
			FROM dbo.tArtikel
			JOIN #Daten_spProductUpdate AS Daten ON Daten.kArtikel = tArtikel.kArtikel;


	--
	-- tArtikelSpeicher aktualisieren
	--

	DECLARE @AktualisiereArtikelspeicher Artikel.TYPE_spAktualisiereArtikelspeicher;
	INSERT INTO @AktualisiereArtikelspeicher
	SELECT DISTINCT #Daten_spProductUpdate.kArtikel FROM #Daten_spProductUpdate 
	EXEC Artikel.spAktualisiereArtikelspeicher @Artikel = @AktualisiereArtikelspeicher -- TYPE_spAktualisiereArtikelspeicher
	

	--
	-- Hier kommt der Code nach dem INSERT/UPDATE/DELETE rein
	--
	

	-- Rückgabe Anzahl der Datensätze
	SELECT @ReturnValue = COUNT(1)
		FROM #Temp_spProductUpdate;
		IF(@CreatedTransaction = 1)
		BEGIN
			COMMIT-- Nur wenn kein Savepoint gesetzt
		END
		RETURN @ReturnValue;
	END TRY
	BEGIN CATCH
	-- Dedlock Catch
		IF(ERROR_NUMBER() = 1205)
		BEGIN
			SET @retry = @retry - 1;
			IF(@CreatedTransaction = 1)
			BEGIN
			    ROLLBACK TRANSACTION;
			END
			ELSE
			BEGIN
			    ROLLBACK TRANSACTION Savepoint0;
			END
			IF(@retry = 0)
			BEGIN
				SET CONTEXT_INFO 0x0;
				THROW;
				RETURN 0;
			END
			EXEC dbo.spWaitRandom;
		END
		ELSE
		BEGIN
			SET @retry = -1;
			IF(@CreatedTransaction = 1)
			BEGIN
			    ROLLBACK TRANSACTION;
			END
			ELSE
			BEGIN
			    ROLLBACK TRANSACTION Savepoint0;
			END
			SET CONTEXT_INFO @OldContextInfo;
			THROW;
			RETURN 0;
		END
	END CATCH
END
