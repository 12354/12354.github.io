CREATE PROCEDURE Maintenance.spStuecklistenAktualisieren
	@kArtikel INT = NULL,
	@cArtNr NVARCHAR(100) = NULL
AS
BEGIN
	DECLARE @Stuecklisten AS Artikel.TYPE_spAktualisiereStueckliste;
	IF(@kArtikel IS NOT NULL)
	BEGIN
		INSERT INTO @Stuecklisten(kArtikel_Stueckliste)
			VALUES (@kArtikel);
	END
	ELSE IF(@cArtNr IS NOT NULL)
	BEGIN
		INSERT INTO @Stuecklisten(kArtikel_Stueckliste)
			SELECT kArtikel
				FROM dbo.tArtikel
				WHERE cArtNr = @cArtNr;
	END
	ELSE
	BEGIN
		INSERT INTO @Stuecklisten(kArtikel_Stueckliste)
			SELECT kArtikel
				FROM Maintenance.vStuecklisten;
	END
	IF(EXISTS(SELECT * FROM @Stuecklisten))
	BEGIN
		EXEC Artikel.spAktualisiereStueckliste @Artikel = @Stuecklisten;		
	END
END
