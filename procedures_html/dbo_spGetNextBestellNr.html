dbo.spGetNextBestellNr

Description: Die Stored Procedure gibt die nächste Bestellnummer zurück mit Hilfe von dbo.spGetNextNummer.

CREATE PROC [dbo].[spGetNextBestellNr] 
AS 
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @cNextNummer NVARCHAR(30)
DECLARE @TableNummer TABLE(Nummer NVARCHAR(30))
DECLARE @retry INT = 5;
WHILE @retry > 0
BEGIN      
    BEGIN TRY  
        DECLARE @TranCount INT = @@TRANCOUNT;
        IF(@TranCount = 0)
        BEGIN
            BEGIN TRANSACTION  
        END
	   INSERT INTO @TableNummer(Nummer) EXEC [spGetNextNummer] @cName = 'Auftrag', @kFirma = 0, @nNoUpdate = 0
	   SET @cNextNummer = (SELECT Nummer FROM @TableNummer)
	   SELECT @cNextNummer
	   
	   SET @retry = 0;
        IF(@TranCount = 0)
        BEGIN           
            COMMIT TRANSACTION      
        END
    END TRY       
    BEGIN CATCH           
        IF(ERROR_NUMBER() = 1205 AND @TranCount = 0 AND XACT_STATE() <> 0)           
        BEGIN              
            SET @retry = @retry - 1;               
            ROLLBACK       
        END          
        ELSE          
        BEGIN              
            SET @retry = 0;               
            THROW;           
        END      
    END CATCH
END
