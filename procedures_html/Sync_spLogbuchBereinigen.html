Sync.spLogbuchBereinigen

Description: Die Procedure löscht veraltete Logbuch Einträge.

Long Description: Keine ausführliche Beschreibung hinterlegt.

Created At: 2024-01-12 03:32:04

Updated At: 2024-01-12 03:32:04

Code:

CREATE PROCEDURE [Sync].[spLogbuchBereinigen]
AS
BEGIN
	DELETE Sync.tSyncLogsitzung
	FROM Sync.tSyncLogsitzung 
	LEFT JOIN dbo.tLoggingConfiguration ON tSyncLogsitzung.nTyp = tLoggingConfiguration.kLogsitzungTyp
	WHERE tSyncLogsitzung.dStart < DATEADD(DAY, -ISNULL(tLoggingConfiguration.nAufbewahrungErfolgreich, 3), GETDATE())     
		AND tSyncLogsitzung.nStatus = 16; --Erfolgreich abgeschlossen


	DELETE Sync.tSyncLogsitzung
	FROM Sync.tSyncLogsitzung 
	LEFT JOIN dbo.tLoggingConfiguration ON tSyncLogsitzung.nTyp = tLoggingConfiguration.kLogsitzungTyp
	WHERE tSyncLogsitzung.dStart < DATEADD(DAY, -ISNULL(tLoggingConfiguration.nAufbewahrungFehlerhaft, 7), GETDATE())     
		AND tSyncLogsitzung.nStatus != 16;	
END;
