CREATE PROCEDURE [CustomWorkflows].[_SetActionParameterDisplayName]    
    @parameterName VARCHAR(128),
    @displayName VARCHAR(128)
AS
BEGIN

    IF EXISTS
    (
        SELECT *
        FROM sys.extended_properties displayNameUserType
            JOIN sys.types userTypes
                ON userTypes.is_user_defined = 1
                   AND displayNameUserType.major_id = userTypes.user_type_id
        WHERE displayNameUserType.name = 'DisplayName'
              AND displayNameUserType.class_desc = 'TYPE'
              AND userTypes.name = @parameterName
    )
    BEGIN

        EXEC sys.sp_dropextendedproperty @name = N'DisplayName',
                                         @level0type = N'SCHEMA',
                                         @level0name = N'CustomWorkflows',
                                         @level1type = N'TYPE',
                                         @level1name = @parameterName;
    END;

    EXEC sys.sp_addextendedproperty @name = N'DisplayName',
                                    @value = @displayName,
                                    @level0type = 'SCHEMA',
                                    @level0name = N'CustomWorkflows',
                                    @level1type = 'TYPE',
                                    @level1name = @parameterName,
                                    @level2type = NULL,
                                    @level2name = NULL;

END;
