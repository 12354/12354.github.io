pps.spUpdateScrapRatio

Description: Mit der Stored Procedure kann für die übergebenen Artikel die Ausschuss Rate neu Berechnet werden.

Long Description: Keine ausführliche Beschreibung hinterlegt.

CREATE PROCEDURE pps.spUpdateScrapRatio @Artikel AS pps.TYPE_spUpdateScrapRatio READONLY
AS
DECLARE @retry INT = 5;
WHILE @retry > 0
BEGIN
    BEGIN TRY
        DECLARE @TranCount INT = @@TRANCOUNT;
        IF (@TranCount = 0)
        BEGIN
            BEGIN TRANSACTION
        END


        IF EXISTS (SELECT 1 FROM @Artikel)
        BEGIN
            UPDATE pps.tProdCondition
            SET tProdCondition.nRejectsRatio = t1.fRatio
            FROM pps.tProdCondition
                JOIN
                (
                    SELECT tProdCondition.kArtikel,
                           tProdCondition.kZustand,
                           AVG(tWorkJobResult.fRatio) AS fRatio
                    FROM pps.tProdCondition
                        JOIN pps.tWorkJob
                            ON tProdCondition.kArtikel = tWorkJob.kArtikel
                        JOIN pps.tWorkJobResult
                            ON tWorkJob.kWorkJob = tWorkJobResult.kWorkJob
                               AND tProdCondition.kZustand = tWorkJobResult.kZustand
                    GROUP BY tProdCondition.kArtikel,
                             tProdCondition.kZustand
                ) AS t1
                    ON t1.kArtikel = tProdCondition.kArtikel
                       AND t1.kZustand = tProdCondition.kZustand
                JOIN @Artikel AS artikel
                    ON artikel.kArtikel = tProdCondition.kArtikel
            WHERE tProdCondition.kZustand <> 1
        END
        ELSE
        BEGIN
            UPDATE pps.tProdCondition
            SET tProdCondition.nRejectsRatio = t1.fRatio
            FROM pps.tProdCondition
                JOIN
                (
                    SELECT tProdCondition.kArtikel,
                           tProdCondition.kZustand,
                           AVG(tWorkJobResult.fRatio) AS fRatio
                    FROM pps.tProdCondition
                        JOIN pps.tWorkJob
                            ON tProdCondition.kArtikel = tWorkJob.kArtikel
                        JOIN pps.tWorkJobResult
                            ON tWorkJob.kWorkJob = tWorkJobResult.kWorkJob
                               AND tProdCondition.kZustand = tWorkJobResult.kZustand
                    GROUP BY tProdCondition.kArtikel,
                             tProdCondition.kZustand
                ) AS t1
                    ON t1.kArtikel = tProdCondition.kArtikel
                       AND t1.kZustand = tProdCondition.kZustand
            WHERE tProdCondition.kZustand <> 1
        END
        SET @retry = 0;
        IF (@TranCount = 0)
        BEGIN
            COMMIT TRANSACTION
        END
    END TRY
    BEGIN CATCH
        IF (ERROR_NUMBER() = 1205 AND @TranCount = 0 AND XACT_STATE() <> 0)
        BEGIN
            SET @retry = @retry - 1;
            ROLLBACK
        END
        ELSE
        BEGIN
            SET @retry = 0;
            THROW;
        END
    END CATCH
END
