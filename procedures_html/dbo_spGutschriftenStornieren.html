CREATE PROCEDURE [dbo].[spGutschriftenStornieren]
	@Gutschriften dbo.TYPE_spGutschriftenStornieren READONLY,
	@kGutschriftStornogrund INT,
	@cKommentar NVARCHAR(100),
	@kBenutzer INT,
	@dStorniert DATETIME,
	@xResult XML OUTPUT
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT ON;
BEGIN TRY
	DECLARE @DeadlockRetries TINYINT = CASE WHEN @@TRANCOUNT = 0 AND XACT_STATE() = 0 THEN 5 ELSE 0 END;
	DECLARE @OldContextInfo VARBINARY(128) = ISNULL(CONTEXT_INFO(), 0x0);
	DECLARE @MyContextInfo VARBINARY(128) = HASHBYTES('SHA1', OBJECT_NAME(@@PROCID));
	SET CONTEXT_INFO @MyContextInfo;
DeadlockRetry:
	BEGIN TRY
		IF (@DeadlockRetries > 0) BEGIN TRANSACTION;

		IF(OBJECT_ID('tempdb..#GutschriftStorno') IS NOT NULL)
		BEGIN
			DROP TABLE #GutschriftStorno;
		END

		CREATE TABLE #GutschriftStorno
		(
			kGutschrift INT PRIMARY KEY NOT NULL,
			cGutschriftNr NVARCHAR(50) NOT NULL,
			nError INT NOT NULL,
			kStornoGutschrift INT NOT NULL
		);

		INSERT INTO #GutschriftStorno (kGutschrift, cGutschriftNr, nError, kStornoGutschrift)
		SELECT DISTINCT [@Gutschriften].kGutschrift, N'', 0, [@Gutschriften].kStornoGutschrift FROM @Gutschriften

		UPDATE #GutschriftStorno
		SET
			#GutschriftStorno.cGutschriftNr = vGutschriftStornierbar.cGutschriftNr,
			#GutschriftStorno.nError = vGutschriftStornierbar.nError
		FROM #GutschriftStorno
		JOIN dbo.vGutschriftStornierbar ON vGutschriftStornierbar.kGutschrift = #GutschriftStorno.kGutschrift

		-- Zahlungen von Gutschriften auf Rechnungen ziehen
		UPDATE dbo.tZahlung
		SET
			tZahlung.kRechnung = tgutschrift.kRechnung,
			tZahlung.kGutschrift = NULL
		FROM dbo.tZahlung
		JOIN #GutschriftStorno ON #GutschriftStorno.kGutschrift = tZahlung.kGutschrift
		JOIN tgutschrift ON #GutschriftStorno.kGutschrift = tgutschrift.kGutschrift
		WHERE #GutschriftStorno.nError = 0

		-- tZahlungsabgleichAusgang setzen
		UPDATE dbo.tZahlungsabgleichAusgang
		SET
			tZahlungsabgleichAusgang.kGutschrift = 0
		FROM dbo.tZahlungsabgleichAusgang
		JOIN #GutschriftStorno ON #GutschriftStorno.kGutschrift = tZahlungsabgleichAusgang.kGutschrift
		WHERE #GutschriftStorno.nError = 0

		INSERT INTO dbo.tGutschriftStorno(kGutschrift, kBenutzer, kGutschriftStornogrund, dStorniert, cKommentar, kStornoGutschrift)
		SELECT #GutschriftStorno.kGutschrift, @kBenutzer, @kGutschriftStornogrund, @dStorniert, @cKommentar, #GutschriftStorno.kStornoGutschrift
		FROM #GutschriftStorno
		WHERE #GutschriftStorno.nError = 0

		UPDATE dbo.tgutschrift
		SET tgutschrift.nStorno = 1
		FROM dbo.tgutschrift
		JOIN #GutschriftStorno ON #GutschriftStorno.kGutschrift = tgutschrift.kGutschrift
		WHERE #GutschriftStorno.nError = 0

		--Eckdaten aktualisieren
		IF EXISTS (SELECT * FROM #GutschriftStorno WHERE #GutschriftStorno.nError = 0)
		BEGIN
			DECLARE @GutschriftenEckdaten dbo.TYPE_spGutschriftEckdatenBerechnen

			INSERT INTO @GutschriftenEckdaten(kGutschrift)
			SELECT #GutschriftStorno.kGutschrift
			FROM #GutschriftStorno

			EXECUTE dbo.spGutschriftEckdatenBerechnen @Gutschriften = @GutschriftenEckdaten
		END

		--
		-- Stornoergebnis zurückmelden
		--
		SET @xResult = (
			SELECT
				#GutschriftStorno.kGutschrift AS kGutschrift,
				#GutschriftStorno.cGutschriftNr AS cGutschriftNr,
				#GutschriftStorno.nError AS nError
			FROM #GutschriftStorno
			FOR XML PATH('StornoResult'), ROOT('StornoResults')
		)


		-- Ende vom Code
		IF (@DeadlockRetries > 0) AND (@@TRANCOUNT > 0) AND (XACT_STATE() = 1) COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		BEGIN TRY
			EXEC spErrorHandler @DeadlockRetries = @DeadlockRetries OUTPUT;
			IF (@DeadlockRetries > 0) GOTO DeadlockRetry;
		END TRY
		BEGIN CATCH
			IF ERROR_NUMBER() <> 266 THROW;
		END CATCH;
		THROW;
	END CATCH

	SET CONTEXT_INFO @OldContextInfo;
END TRY
BEGIN CATCH
	SET CONTEXT_INFO @OldContextInfo;
	THROW;
END CATCH
