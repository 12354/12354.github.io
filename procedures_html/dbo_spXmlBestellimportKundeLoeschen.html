CREATE PROCEDURE dbo.spXmlBestellimportKundeLoeschen
	@kKunde INT
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
BEGIN
	DECLARE @OldContextInfo VARBINARY(128);  
	IF(CONTEXT_INFO() IS NOT NULL)   
	BEGIN    
		SET @OldContextInfo = CONTEXT_INFO();   
	END   
	ELSE   
	BEGIN    
		SET @OldContextInfo = 0x0;   
	END   
	DECLARE @hash VARBINARY(128);   
	SELECT @hash = HASHBYTES('SHA1', 'dbo.spXmlBestellimportKundeLoeschen');   
	SET CONTEXT_INFO @hash; 

	DELETE FROM dbo.tkunde
	WHERE tkunde.kKunde = @kKunde;

	SET CONTEXT_INFO @OldContextInfo;     
END
