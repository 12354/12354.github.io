CREATE PROCEDURE [dbo].[spZulaufShopAktualisieren]
--
-- Die SP füllt die Felder dZulaufVerfuegbarAm und nZulaufVerfuegbarAm in tArtikel
-- 
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
BEGIN
    DECLARE @cOptionZuletztAktualisiert AS VARCHAR(30);
    SET @cOptionZuletztAktualisiert = 'ShopZulaufZuletztAktualisiert';
 
    --
    -- Prüfen ob Aktualisierung überhaupt erfolgen soll
    --
    IF(ISNULL((SELECT cValue FROM dbo.tOptions WHERE cKey = 'ShopZulaufAktualisieren'), 1) = 0)
    BEGIN
        RETURN;
    END;
 
    --
    -- Prüfen ob Aktualisierung immer erfolgen soll
    --
    IF(ISNULL((SELECT cValue FROM dbo.tOptions WHERE cKey = 'ShopZulaufImmerAktualisieren'), 1) = 0)
    BEGIN
        --
        -- Prüfen ob die letzte Aktualisierung länger als 24 Stunden her ist
        --
        IF((SELECT DATEDIFF(hour, CONVERT(DATETIME, (SELECT ISNULL((SELECT cValue FROM dbo.tOptions WHERE cKey = @cOptionZuletztAktualisiert), CONVERT(VARCHAR(20), DATEADD(day, -1, GETDATE())))), 102), GETDATE())) < 24)
        BEGIN
            RETURN;
        END;
    END;
 
    -- Zuläufe für die Lagerartikel in den tArtikel schreiben (inkl. Zulauf löschen, deshalb der LEFT JOIN)
    DECLARE @tArtikel AS TABLE (kArtikel INT NOT NULL);

    UPDATE dbo.tArtikel
    SET dZulaufVerfuegbarAm = vNaechsterZulaufLagerartikel.dLieferdatum,
        nZulaufVerfuegbarMenge = ISNULL(vNaechsterZulaufLagerartikel.fAnzahlZulauf, 0)
    OUTPUT INSERTED.kArtikel
    INTO @tArtikel
    FROM dbo.tArtikel
    LEFT JOIN Abgleich.vNaechsterZulaufLagerartikel ON vNaechsterZulaufLagerartikel.kArtikel = tArtikel.kArtikel
    WHERE	(tArtikel.kStueckliste = 0 OR tArtikel.kStueckliste IS NULL)
			AND
			(
				CONVERT(DATE, tArtikel.dZulaufVerfuegbarAm) <> vNaechsterZulaufLagerartikel.dLieferdatum
				OR tArtikel.nZulaufVerfuegbarMenge <> ISNULL(vNaechsterZulaufLagerartikel.fAnzahlZulauf, 0)
			);
 
    -- Zuläufe für die Stücklistenväter in den tArtikel schreiben
	UPDATE dbo.tArtikel
	SET tArtikel.dZulaufVerfuegbarAm = vLagerbestandStueckliste.dLieferdatum,
		tArtikel.nZulaufVerfuegbarMenge = ISNULL(vLagerbestandStueckliste.fZulauf, 0)
	FROM 
	(
		SELECT DISTINCT tStueckliste.kStueckliste
		FROM @tArtikel AS tArtikel
		JOIN dbo.tStueckliste ON tStueckliste.kArtikel = tArtikel.kArtikel
	) AS Stueckliste
	JOIN dbo.tArtikel ON tArtikel.kStueckliste = Stueckliste.kStueckliste
	JOIN dbo.vLagerbestandStueckliste ON vLagerbestandStueckliste.kStueckliste = Stueckliste.kStueckliste
	WHERE	(tArtikel.kStueckliste <> 0)
			AND 
			(
				CONVERT(DATE, tArtikel.dZulaufVerfuegbarAm) <> vLagerbestandStueckliste.dLieferdatum
				OR tArtikel.nZulaufVerfuegbarMenge <> ISNULL(vLagerbestandStueckliste.fZulauf, 0)
			);
 
    --
    -- Letzte Aktualisierung in tOptions schreiben
    --
    MERGE INTO dbo.tOptions AS Ziel
    USING (SELECT CONVERT(VARCHAR, GETDATE(), 102) AS Zeitstempel) AS Source
    ON (Ziel.cKey = @cOptionZuletztAktualisiert)
    WHEN MATCHED THEN
        UPDATE SET Ziel.cValue = Source.Zeitstempel
    WHEN NOT MATCHED BY TARGET THEN
        INSERT (cKey, cValue)
        VALUES (@cOptionZuletztAktualisiert, Source.Zeitstempel);
 
END;
