CREATE PROCEDURE [Auslieferung].[spPicklistenVerwerfen]


-- Enthält den Benutzer zur eindeutigen Zuordnung
@kBenutzer INT,

@kSessionId INT

AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

DELETE FROM dbo.tPickliste
    WHERE dbo.tPickliste.kSessionId = @kSessionId AND dbo.tPickliste.nStatus < 10
