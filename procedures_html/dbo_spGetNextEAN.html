CREATE PROCEDURE [dbo].[spGetNextEAN]
	@AmmountOfEans INT
AS
DECLARE @retry INT = 5;
WHILE @retry > 0
BEGIN 

	DECLARE @CurrentTimeStamp datetime = CURRENT_TIMESTAMP
	IF(OBJECT_ID('tempdb..#UsedEans') IS NOT NULL)
	BEGIN
		DROP TABLE #UsedEans
	END;
	CREATE TABLE #UsedEans
	(   
		cEAN VARCHAR(255) 
	);
	
	BEGIN TRY  
        DECLARE @TranCount INT = @@TRANCOUNT;
        IF(@TranCount = 0)
        BEGIN
            BEGIN TRANSACTION  
        END

	   INSERT INTO #UsedEans 
	   SELECT TOP (@AmmountOfEans)	tEAN.cEAN
	   FROM dbo.tEAN
	   WHERE NOT EXISTS
	   (
		  SELECT 1 
		  FROM dbo.tGebinde 
		  WHERE tEAN.cEAN = tGebinde.cEAN
	   )
	   AND NOT EXISTS
	   (
		  SELECT 1 
		  FROM dbo.tArtikel 
		  WHERE tEAN.cEAN = tArtikel.cBarcode
	   )
	   AND 
	   (
		  tEAN.dLockDate IS NULL 
		  OR CONVERT(Datetime, tEAN.dLockDate, 104) < CONVERT(Datetime, DATEADD(D, -1, @CurrentTimeStamp), 104)
	   )
	   ORDER BY tEAN.cEAN

	   UPDATE dbo.tEAN
	   SET tEAN.dLockDate = @CurrentTimeStamp 
	   WHERE tEAN.cEAN IN 
	   (
		  SELECT *
		  FROM #UsedEans 
	   )

	   SET @retry = 0;
        IF(@TranCount = 0)
        BEGIN           
            COMMIT TRANSACTION      
        END
    END TRY       
    BEGIN CATCH           
        IF(ERROR_NUMBER() = 1205 AND @TranCount = 0 AND XACT_STATE() <> 0)           
        BEGIN              
            SET @retry = @retry - 1;               
            ROLLBACK       
        END          
        ELSE          
        BEGIN              
            SET @retry = 0;               
            THROW;           
        END      
    END CATCH

END

SELECT * 
FROM #UsedEans
