CREATE PROCEDURE SCX.spLoescheKaputteExterneRechnung 
	@kRechnung INT,
	@kBenutzer INT,
	@LogMessage NVARCHAR(4000)
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT ON;
BEGIN TRY
	DECLARE @DeadlockRetries TINYINT = CASE WHEN @@TRANCOUNT = 0 AND XACT_STATE() = 0 THEN 5 ELSE 0 END;
	
	IF ISNULL(@kRechnung, 0) = 0 OR ISNULL(@kBenutzer, 0) = 0
	BEGIN
		RETURN;
	END
	
	IF NOT EXISTS (SELECT
					tRechnung.kRechnung
				FROM Rechnung.tRechnung
				JOIN dbo.tShop ON tShop.kShop = tRechnung.kShop
				LEFT JOIN dbo.tgutschrift ON tgutschrift.kRechnung = tRechnung.kRechnung
				WHERE tRechnung.kRechnung = @kRechnung
					AND tgutschrift.kGutschrift IS NULL
					AND tShop.nTyp = 6
					AND tRechnung.nIstExterneRechnung > 0)
	BEGIN
		RETURN;
	END

	DeadlockRetry:
    BEGIN TRY
        IF (@DeadlockRetries > 0) BEGIN TRANSACTION;

		IF EXISTS(SELECT 1 FROM dbo.tZahlung WHERE tZahlung.kRechnung = @kRechnung)
		BEGIN
			DELETE FROM dbo.tZahlung
			WHERE tZahlung.kRechnung = @kRechnung;
		END

		UPDATE Rechnung.tRechnung
		SET tRechnung.nIstEntwurf = 1
		WHERE tRechnung.kRechnung = @kRechnung;
	
		EXEC Rechnung.spRechnungLoeschen @RechnungKey = @kRechnung,
										 @BenutzerKey = 1,
										 @LogMessage = @LogMessage,
										 @OnlyValidate = 0;

		-- Ende vom Code
        IF (@DeadlockRetries > 0)
           AND (@@TRANCOUNT > 0)
           AND (XACT_STATE() = 1) COMMIT TRANSACTION;
	END TRY
    BEGIN CATCH
        BEGIN TRY
            EXEC dbo.spErrorHandler @DeadlockRetries = @DeadlockRetries OUTPUT;
            IF (@DeadlockRetries > 0) GOTO DeadlockRetry;
        END TRY
        BEGIN CATCH
            IF ERROR_NUMBER() <> 266 THROW;
        END CATCH;
        THROW;
    END CATCH;
END TRY
BEGIN CATCH
    THROW;
END CATCH;
