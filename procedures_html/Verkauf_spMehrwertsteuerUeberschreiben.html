CREATE PROCEDURE [Verkauf].[spMehrwertsteuerUeberschreiben]
@AuftragKey INT,
@MwStSatzAlt DECIMAL(28,14),
@MwStSatzNeu DECIMAL(28,14),
@BruttoNeuRechnen INT
AS
BEGIN

IF EXISTS(
	SELECT 1
	FROM Verkauf.tAuftragEckdaten
	WHERE tAuftragEckdaten.kAuftrag = @AuftragKey AND tAuftragEckdaten.nRechnungStatus > 0
)
BEGIN
	RAISERROR (N'Der Auftrag wurde bereits abgerechnet. Die Änderung der Mehrwertsteuer ist nicht mehr möglich.', 18, 1);
	RETURN;
END

IF(@BruttoNeuRechnen = 1)
BEGIN
	UPDATE Verkauf.tAuftragPosition
	SET fMwSt = @MwStSatzNeu
	WHERE tAuftragPosition.kAuftrag = @AuftragKey AND tAuftragPosition.fMwSt = @MwStSatzAlt
END
ELSE
BEGIN
	UPDATE Verkauf.tAuftragPosition
	SET fVkNetto = tAuftragPosition.fVkNetto * (@MwStSatzAlt / 100 + 1.0) / (@MwStSatzNeu / 100 + 1.0),
		fMwSt = @MwStSatzNeu
	WHERE tAuftragPosition.kAuftrag = @AuftragKey AND tAuftragPosition.fMwSt = @MwStSatzAlt
END

--Eckdaten neu berechnen
DECLARE @Auftrag Verkauf.TYPE_spAuftragEckdatenBerechnen

INSERT INTO @Auftrag (kAuftrag)
VALUES(@AuftragKey)

EXEC Verkauf.spAuftragEckdatenBerechnen @Auftrag

END
