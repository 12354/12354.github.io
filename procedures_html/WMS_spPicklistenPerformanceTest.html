CREATE PROCEDURE [WMS].[spPicklistenPerformanceTest]
@kWarenlager INT = 9

--
-- Diese SP führt versucht alle Picklistenvorlagen in der DB zu reservieren und gibt Statistiken über die Reservierungszeit wieder.
-- Die SP löscht alle nicht abgeschloßenen Picklisten, auf keinen Fall Produktiv verwenden.
-- !!! DIESE SP WIRD DARF NICHT PRODUKTIV VERDENDET !!!!
--


AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

	
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @kPickliste INT;
DECLARE @kPicklisteVorlage INT;
DECLARE @kBenutzer INT = 1;
DECLARE @kSessionID INT = 0;
DECLARE @nDebug INT = 0;
DECLARE @kRollendeKommissionierungWagen INT = 0;
DECLARE @dStartTime DATETIME;
DECLARE @dEndTime DATETIME;
DECLARE @nAnzahlAuftraege INT;
DECLARE @Plan XML;
DECLARE @nLastPickliste INT = 0;
DECLARE @dTimeStamp DATETIME = GETDATE();

BEGIN TRY

	DBCC FREEPROCCACHE;


	DECLARE CUR_PicklisteVorlagen  CURSOR FAST_FORWARD LOCAL FOR
	SELECT tPicklisteVorlage.kPicklisteVorlage 
	FROM dbo.tPicklisteVorlage
	WHERE nIstVorlage = 1


	IF(object_id('tempdb..#PicklistenStatistik') IS NOT NULL)
	BEGIN
		DROP TABLE #PicklistenStatistik;
	END;
	CREATE TABLE #PicklistenStatistik(kPicklisteVorlage INT,dStartZeit DATETIME, dEndZeit DATETIME, nAnzahlAuftraege INT);


	SELECT @nLastPickliste = MAX(kPickliste)
	FROM dbo.tPickliste;
	

	
	OPEN CUR_PicklisteVorlagen;
	FETCH NEXT FROM CUR_PicklisteVorlagen INTO @kPicklisteVorlage

	WHILE (@@FETCH_STATUS = 0 ) 
	BEGIN

		SET @dStartTime = GETDATE();

		EXECUTE [WMS].[spPicklisteErstellen] 
		   @kWarenlager
		  ,@kPicklisteVorlage
		  ,@kBenutzer
		  ,@kSessionID
		  ,@nDebug
		  ,@kRollendeKommissionierungWagen

		  SET @dEndTime = GETDATE();


		  SELECT @nAnzahlAuftraege = COUNT(DISTINCT kBestellung) FROM tpicklistepos WHERE tPicklistePos.nStatus < 40;
		  INSERT INTO #PicklistenStatistik VALUES (@kPicklisteVorlage,@dStartTime,@dEndTime,@nAnzahlAuftraege);

		  DELETE FROM dbo.tPicklistePos
		  WHERE tPicklistePos.nStatus < 40
		  AND tPicklistePos.kPickliste > @nLastPickliste;

	FETCH NEXT FROM CUR_PicklisteVorlagen INTO @kPicklisteVorlage
	END;

	CLOSE CUR_PicklisteVorlagen;
	DEALLOCATE CUR_PicklisteVorlagen;

	SELECT tPicklisteVorlage.cName, #PicklistenStatistik.kPicklisteVorlage  ,dStartZeit , dEndZeit , nAnzahlAuftraege , DATEDIFF(SECOND,dStartZeit,dEndZeit) AS Sekunden ,DATEDIFF(MILLISECOND,dStartZeit,dEndZeit) AS MilliSec 
	FROM #PicklistenStatistik
	JOIN dbo.tPicklisteVorlage ON tPicklisteVorlage.kPicklisteVorlage = #PicklistenStatistik.kPicklisteVorlage;

	
	INSERT INTO WMS.tStatistikPicklisten(dTimeStamp , cName , kPicklisteVorlage , dStartZeit , dEndZeit , nAnzahlAuftraege ,dSekunden ,dMilliSec )
	SELECT @dTimeStamp,tPicklisteVorlage.cName, #PicklistenStatistik.kPicklisteVorlage  ,dStartZeit , dEndZeit , nAnzahlAuftraege , DATEDIFF(SECOND,dStartZeit,dEndZeit) AS Sekunden ,DATEDIFF(MILLISECOND,dStartZeit,dEndZeit) AS MilliSec 
	FROM #PicklistenStatistik
	JOIN dbo.tPicklisteVorlage ON tPicklisteVorlage.kPicklisteVorlage = #PicklistenStatistik.kPicklisteVorlage;

		

END TRY

BEGIN CATCH

		    DELETE FROM dbo.tPicklistePos
		    WHERE tPicklistePos.nStatus < 40
		    AND tPicklistePos.kPickliste > @nLastPickliste;

			SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();

			RAISERROR (	@ErrorMessage, 
						@ErrorSeverity,
						@ErrorState);
		
END CATCH;
