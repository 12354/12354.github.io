CREATE PROCEDURE [Auslieferung].[spPicklistenUebernehmen]

-- Mögliche Fehler:
-- Im Fehlerfall enthält der Messagetext XML-Informationen zum Fehler
-- Code = 50000, Severity = 18, State = 1: Die übergebene Pickliste würde eine Teillieferung erzeugen, 
--      obwohl Teillieferungen verboten sind. 
-- Code = 50000, Severity = 18, State = 2: Es wurden Pickpositionen übergeben die bereits ausgeliefert wurden
--
-- Transaction-Handling: siehe https://msdn.microsoft.com/en-us/library/ms188378.aspx
--
-- <Bestellung>
--   <kBestellung>7</kBestellung>
-- </Bestellung>
@Bestellungen XML,

-- Enthält den Benutzer zur eindeutigen Zuordnung
@kBenutzer INT,

-- Gibt an ob die Aufträge gesplittet werden sollen (0 = nein, 1 = teillieferung ohne Split, 2 = teillieferung mit split)
@nTeillieferung BIT,

@kSessionId INT,

@xResult XML OUTPUT

AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

DECLARE @TranCounter INT;
SET @TranCounter = @@TRANCOUNT;
IF @TranCounter > 0
	SAVE TRANSACTION spPicklistenUebernehmen
ELSE
	BEGIN TRANSACTION;

BEGIN TRY
    DECLARE @xErrorInfo XML;
	DECLARE @cError NVARCHAR(MAX);
    DECLARE @dNow AS DATETIME;
    SET @dNow = (SELECT GETDATE());
	SET @xResult = '';

	IF(object_id('tempdb..#XMLBESTELLUNGEN') IS NOT NULL)
	BEGIN
	   DROP TABLE #XMLBESTELLUNGEN;
	END
    CREATE TABLE #XMLBESTELLUNGEN(
		kBestellung INT
		)
	INSERT INTO #XMLBESTELLUNGEN
	(
	    #XMLBESTELLUNGEN.kBestellung
	)
	SELECT
		ParamValues.ID.value('kBestellung[1]', 'INT') AS kBestellung
	FROM @Bestellungen.nodes('/Bestellung') AS ParamValues(ID)


    IF (@nTeillieferung = 0)
    BEGIN		
		SET @xErrorInfo = (SELECT
				vAuftrag.kAuftrag AS kBestellung,
				vAuftrag.cAuftragsNr AS cBestellNr,
				COUNT(CASE WHEN vAuftragPositionEckdaten.fAnzahlOffen > 0 THEN 1 END) AS fAnzahlOffenePositionen,
				SUM(vAuftragPositionEckdaten.fAnzahlOffen) AS fAnzahlOffen,
				SUM(ISNULL(PP.fAnzahl, 0)) AS fAnzahlAufPicklisten
				FROM Verkauf.vAuftrag
				JOIN Verkauf.vAuftragPosition ON vAuftrag.kAuftrag = vAuftragPosition.kAuftrag
				JOIN #XMLBESTELLUNGEN ON #XMLBESTELLUNGEN.kBestellung = vAuftrag.kAuftrag
				JOIN Verkauf.vAuftragPositionEckdaten ON vAuftragPositionEckdaten.kAuftragPosition = vAuftragPosition.kAuftragPosition
				LEFT JOIN (
					SELECT
						dbo.tPicklistePos.kBestellPos,
						SUM(dbo.tPicklistePos.fAnzahl) AS fAnzahl
					FROM 
						dbo.tPicklistePos
					GROUP BY
						dbo.tPicklistePos.kBestellPos
					) AS PP ON PP.kBestellPos = vAuftragPosition.kAuftragPosition
				GROUP BY vAuftrag.kAuftrag, vAuftrag.cAuftragsNr
				HAVING SUM(vAuftragPositionEckdaten.fAnzahlOffen) - SUM(ISNULL(PP.fAnzahl, 0)) > 0 AND SUM(ISNULL(PP.fAnzahl, 0)) > 0
			FOR XML PATH('BrokenOrders'), TYPE)
		IF (@xErrorInfo IS NOT NULL)
		BEGIN			
			SET @cError = CAST(@xErrorInfo AS NVARCHAR(MAX))
			RAISERROR(@cError, 18, 1);
		END
    END
		
	-- Überflüssige Stücklistenväter löschen
	DELETE dbo.tPicklistePos
	WHERE dbo.tPicklistePos.kPicklistePos IN
	(
		SELECT dbo.tPicklistePos.kPicklistePos
		FROM dbo.tPicklistePos
		JOIN dbo.tPickliste ON dbo.tPickliste.kPickliste = dbo.tPicklistePos.kPickliste AND dbo.tPickliste.kSessionId = @kSessionId
		JOIN Verkauf.vAuftragPosition AS vaterArtikel ON dbo.tPicklistePos.kBestellPos = vaterartikel.kAuftragPosition
			AND vaterArtikel.kAuftragPosition = vaterArtikel.kAuftragStueckliste
		JOIN Verkauf.vAuftragPosition AS kindArtikel ON vaterArtikel.kAuftragStueckliste = kindArtikel.kAuftragStueckliste
			AND kindArtikel.kAuftragPosition <> kindArtikel.kAuftragStueckliste
		LEFT JOIN dbo.tPicklistePos AS gepickteKindArtikel ON kindArtikel.kAuftragPosition = gepickteKindArtikel.kBestellPos
			AND dbo.tPicklistePos.kPickliste = gepickteKindArtikel.kPickliste
		GROUP BY dbo.tPicklistePos.kPicklistePos
		HAVING MAX(ISNULL(gepickteKindArtikel.kPicklistePos, 0)) = 0
	)

    -- leere Picklisten löschen
    DELETE dbo.tPickliste
    FROM dbo.tPickliste
    LEFT JOIN dbo.tPicklistePos ON dbo.tPicklistePos.kPickliste = dbo.tPickliste.kPickliste
    WHERE dbo.tPicklistePos.kPickliste IS NULL AND dbo.tPickliste.kSessionId = @kSessionId

    -- Status der Positionen hochsetzen
    INSERT INTO dbo.tPicklistePosStatus
    (
        dbo.tPicklistePosStatus.kPicklistePos,
        dbo.tPicklistePosStatus.kbenutzer,
        dbo.tPicklistePosStatus.dZeitstempel,
        dbo.tPicklistePosStatus.nStatus
    )
    SELECT 
        dbo.tPicklistePos.kPicklistePos,
        @kBenutzer,
        @dNow,
        10
    FROM dbo.tPicklistePos 
    JOIN dbo.tPickliste ON dbo.tPickliste.kPickliste = dbo.tPicklistePos.kPickliste AND dbo.tPickliste.kSessionId = @kSessionId
	WHERE dbo.tPicklistePos.nStatus < 10

	DECLARE @OldContextInfo VARBINARY(128);
	IF(CONTEXT_INFO() IS NOT NULL)
	BEGIN
		SET @OldContextInfo = CONTEXT_INFO();
	END
	ELSE
	BEGIN
		SET @OldContextInfo = 0x0;
	END
	
	SET CONTEXT_INFO 0x5022;

	CREATE TABLE #tempPicklistenPosKeys(kPicklistePos INT);

	INSERT INTO #tempPicklistenPosKeys(kPicklistePos)
	SELECT 
		tPicklistePos.kPicklistePos
	FROM dbo.tPicklistePos 
	JOIN dbo.tPickliste ON tPickliste.kPickliste = tPicklistePos.kPickliste AND tPickliste.kSessionId = @kSessionId
	JOIN dbo.tbestellpos ON tbestellpos.kBestellPos = tPicklistePos.kBestellPos
	WHERE tPicklistePos.nStatus <= 10 AND tbestellpos.nType IN(18, 20)

	INSERT INTO dbo.tPicklistePosStatus
	(
		tPicklistePosStatus.kPicklistePos,
		tPicklistePosStatus.kbenutzer,
		tPicklistePosStatus.dZeitstempel,
		tPicklistePosStatus.nStatus
	)
	SELECT 
		#tempPicklistenPosKeys.kPicklistePos,
		@kBenutzer,
		@dNow,
		20
	FROM #tempPicklistenPosKeys; 

	UPDATE dbo.tPicklistePos
	SET nStatus = 20
	FROM dbo.tPicklistePos
	JOIN #tempPicklistenPosKeys ON #tempPicklistenPosKeys.kPicklistePos = tPicklistePos.kPicklistePos
	
	DROP TABLE #tempPicklistenPosKeys;

	SET CONTEXT_INFO @OldContextInfo;

	SET @xResult = (
				SELECT
					(SELECT						
							kPickliste
						FROM
							dbo.tPickliste
						WHERE dbo.tPickliste.kSessionId = @kSessionId
						FOR XML PATH('Pickliste'), ROOT('NewPicklisten'), ELEMENTS, TYPE
					)
				FOR XML PATH('Result')
			)

	IF @TranCounter = 0
		COMMIT TRANSACTION;
END TRY
BEGIN CATCH
	IF @TranCounter = 0
		ROLLBACK TRANSACTION;
	ELSE
		IF XACT_STATE() <> -1
			ROLLBACK TRANSACTION spPicklistenUebernehmen;

	DECLARE @ErrorMessage NVARCHAR(MAX);
    DECLARE @ErrorSeverity INT;
    DECLARE @ErrorState INT;

    SELECT @ErrorMessage = ERROR_MESSAGE();
    SELECT @ErrorSeverity = ERROR_SEVERITY();
    SELECT @ErrorState = ERROR_STATE();

    RAISERROR (@ErrorMessage, -- Message text.
               @ErrorSeverity, -- Severity.
               @ErrorState -- State.
              );
END CATCH
