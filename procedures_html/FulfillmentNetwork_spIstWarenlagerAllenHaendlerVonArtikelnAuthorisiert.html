CREATE PROC [FulfillmentNetwork].[spIstWarenlagerAllenHaendlerVonArtikelnAuthorisiert]
    @artikel AS [FulfillmentNetwork].[TYPE_spIstWarenlagerAllenHaendlerVonArtikelnAuthorisiert] READONLY,
    @kWarenlagerZiel AS INT,
	@ok TINYINT = 0 OUTPUT
AS
IF EXISTS
(
    SELECT *
    FROM @artikel AS Artikel
        JOIN FulfillmentNetwork.tProductRef AS ProductRef
            ON ProductRef.kArtikel = Artikel.kArtikel AND ProductRef.nType = 2
        JOIN dbo.tliefartikel AS LiefArtikel
            ON LiefArtikel.tArtikel_kArtikel = ProductRef.kArtikel
               AND LiefArtikel.nStandard = 1
        LEFT JOIN FulfillmentNetwork.tLieferantWarenlager AS WarenlagerZuordnung
            ON WarenlagerZuordnung.kLieferant = LiefArtikel.tLieferant_kLieferant
               AND WarenlagerZuordnung.kWarenlager = @kWarenlagerZiel
   WHERE WarenlagerZuordnung.kWarenlager IS NULL
)
BEGIN
    SET @ok = 0;
END;
ELSE
BEGIN
    SET @ok = 1;
END;
