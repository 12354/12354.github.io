CREATE PROCEDURE [dbo].[spAktiveMandanten]
AS
BEGIN
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @query NVARCHAR(MAX);
IF(object_id('tempdb..#AktiveMandanten') IS NOT NULL)
BEGIN
	DROP TABLE #AktiveMandanten;
END

CREATE TABLE #AktiveMandanten (kMandant INT NOT NULL, cName NVARCHAR(255) COLLATE database_default, cDB NVARCHAR(20) COLLATE database_default);
SET @query = REPLACE((
	SELECT 
		'INSERT INTO #AktiveMandanten (kMandant, cName, cDB) SELECT ' + CONVERT(NVARCHAR(20), tMandant.kMandant) + ' AS kMandant, ''' + REPLACE(tMandant.cName, '''','''''') + ''' AS cName, ''' + tMandant.cDB + ''' AS cDB FROM [' + tMandant.cDB + '].[dbo].[tHintergrundDienst] CROSS JOIN [eazybusiness].[dbo].[tversion] AS HauptVersion CROSS JOIN [' + tMandant.cDB + '].[dbo].[tversion] AS MandantVersion WHERE HauptVersion.cVersion = MandantVersion.cVersion AND tHintergrundDienst.nAktiv > 0;'
	FROM eazybusiness.dbo.tMandant
	JOIN sys.databases ON databases.name = tMandant.cDB COLLATE database_default
	FOR XML PATH ('')), '>', '>');
EXEC sp_sqlexec @query;
SELECT DISTINCT * FROM #AktiveMandanten;
DROP TABLE #AktiveMandanten;
END
