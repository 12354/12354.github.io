CREATE PROCEDURE [FulfillmentNetwork].[spMerchantProductUpdateQueue]
    @Attribute AS [TYPE_spUpdateMerchantProductUpdateQueue] READONLY
AS
INSERT INTO FulfillmentNetwork.tMerchantProductUpdateQueue (kArtikel, nProcessing, dCreatedAt)
(SELECT tArtikelAttribut.kArtikel,
       0 AS nProcessing,
	   SYSDATETIMEOFFSET () AS dCreatedAt       
FROM dbo.tArtikelAttribut
	JOIN @Attribute ON [@Attribute].kAttribut = tArtikelAttribut.kAttribut
    LEFT JOIN FulfillmentNetwork.tMerchantProductUpdateQueue ON tMerchantProductUpdateQueue.kArtikel = tArtikelAttribut.kArtikel
	LEFT JOIN FulfillmentNetwork.tProductRef ON tProductRef.kArtikel = tArtikelAttribut.kArtikel
WHERE (FulfillmentNetwork.tMerchantProductUpdateQueue.kArtikel IS NULL OR nProcessing = 1) AND FulfillmentNetwork.tProductRef.kArtikel IS NOT NULL
GROUP BY tArtikelAttribut.kArtikel)
