CREATE PROCEDURE Steuern.spSteuerdatenCacheAktualisieren
AS
BEGIN
	SET NOCOUNT ON;
	SET ANSI_NULLS ON;
	SET XACT_ABORT ON;
	BEGIN
	    DECLARE @DeadlockRetries TINYINT = CASE WHEN @@TRANCOUNT = 0 AND XACT_STATE() = 0 THEN 5 ELSE 0 END;
	DeadlockRetry:
	    BEGIN TRY
	        IF (@DeadlockRetries > 0) BEGIN TRANSACTION;
	        -- Start vom Code

			TRUNCATE TABLE Steuern.tSteuerdatenCache;

			INSERT INTO Steuern.tSteuerdatenCache (kFirma, cVersandlandIso, cZiellandIso, cZielbundesland, kSteuerzone)
			SELECT DISTINCT
				vSteuerdaten.kFirma,
				vSteuerdaten.cVersandlandIso,
				vSteuerdaten.cZiellandIso,
				vSteuerdaten.cZielbundesland,
				vSteuerdaten.kSteuerzone
			FROM Steuern.vSteuerdaten;

	        -- Ende vom Code
	        IF (@DeadlockRetries > 0) AND (@@TRANCOUNT > 0) AND (XACT_STATE() = 1) COMMIT TRANSACTION;
	    END TRY
	    BEGIN CATCH
	        BEGIN TRY
	            EXEC dbo.spErrorHandler @DeadlockRetries = @DeadlockRetries OUTPUT;
	            IF (@DeadlockRetries > 0) GOTO DeadlockRetry;
	        END TRY
	        BEGIN CATCH
	            IF ERROR_NUMBER() <> 266 THROW;
	        END CATCH;
	        THROW;
	    END CATCH

	END
END
