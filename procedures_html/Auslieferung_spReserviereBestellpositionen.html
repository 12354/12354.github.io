CREATE PROCEDURE [Auslieferung].[spReserviereBestellpositionen]  

-- Reservierung mehrerer Bestellpositionen  
--
-- Transaction-Handling: siehe https://msdn.microsoft.com/en-us/library/ms188378.aspx
--

@Bestellpositionen XML,
-- <Bestellposition>
--   <kBestellPos>1</kBestellPos>
--   <fAnzahl>1.234</fAnzahl>
-- </Bestellposition>
-- <Bestellposition>
--   <kBestellPos>1</kBestellPos>
--   <fAnzahl>1.234</fAnzahl>
-- </Bestellposition>

@Laeger XML,
-- <Warenlagereingang>
--   <kWarenlagereingang>17</kWarenlagereingang>
--   <fAnzahl>2.0</fAnzahl>
--   <nPrio>0</nPrio>
-- </Warenlagereingang>
-- <Warenlagereingang>
--   <kWarenlagereingang>18</kWarenlagereingang>
--   <fAnzahl>2.0</fAnzahl>
--   <nPrio>1</nPrio>
-- </Warenlagereingang>
@Warenlagereingaenge XML,

-- Optionen für die Auslieferung (Bitmaske)
-- 0x0001 = Pro Bestellung eine Pickliste anlegen
-- 0x0100 = Stücklistenkorrektur vornehmen
@nOptions INT,

@kBenutzer INT,

-- SitzungsId
@kSessionId INT

AS  
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;


DECLARE @TranCounter INT;
SET @TranCounter = @@TRANCOUNT;
IF @TranCounter > 0
	SAVE TRANSACTION spReserviereBestellpositionen
ELSE
	BEGIN TRANSACTION;

BEGIN TRY
    DECLARE @kBestellPos AS INT;
    DECLARE @fAnzahl AS DECIMAL(28,14);

    DECLARE cBestellpos CURSOR LOCAL FAST_FORWARD FOR
        SELECT 
           ParamValues.ID.value('kBestellPos[1]', 'INT'),      
           ParamValues.ID.value('fAnzahl[1]', 'DECIMAL(25,13)')
        FROM @Bestellpositionen.nodes('/Bestellposition') AS ParamValues(ID);
	OPEN cBestellpos;
	FETCH NEXT FROM cBestellpos INTO @kBestellPos, @fAnzahl;
	WHILE @@FETCH_STATUS = 0
	BEGIN
		EXEC Auslieferung.spReserviereBestellposition
			@kBestellPos = @kBestellPos,
			@fAnzahl = @fAnzahl,
			@nOptions = @nOptions,
			@Laeger = @Laeger,
			@Warenlagereingaenge = @Warenlagereingaenge,
			@kBenutzer = @kBenutzer,
			@kSessionId = @kSessionId;
		FETCH NEXT FROM cBestellpos INTO @kBestellPos, @fAnzahl;
	END
	CLOSE cBestellpos;
	DEALLOCATE cBestellpos;

	IF ((@nOptions & 0x0100) > 0)
	BEGIN
     	EXEC [Auslieferung].[spPicklistenStuecklistenFix]
		  @kBenutzer = @kBenutzer,
		  @kSessionId = @kSessionId
     END
	IF @TranCounter = 0
		COMMIT TRANSACTION;
END TRY
BEGIN CATCH
	IF @TranCounter = 0
		ROLLBACK TRANSACTION;
	ELSE
		IF XACT_STATE() <> -1
			ROLLBACK TRANSACTION spReserviereBestellpositionen;

	DECLARE @ErrorMessage NVARCHAR(4000);
    DECLARE @ErrorSeverity INT;
    DECLARE @ErrorState INT;

    SELECT @ErrorMessage = ERROR_MESSAGE();
    SELECT @ErrorSeverity = ERROR_SEVERITY();
    SELECT @ErrorState = ERROR_STATE();

    RAISERROR (@ErrorMessage, -- Message text.
               @ErrorSeverity, -- Severity.
               @ErrorState -- State.
              );
END CATCH
