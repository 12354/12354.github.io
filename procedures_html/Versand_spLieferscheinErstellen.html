CREATE PROCEDURE [Versand].[spLieferscheinErstellen]
	@xLieferschein XML = NULL,
	@kBestellung INT,
	@kBenutzer INT,
	@cLieferscheinNr NVARCHAR(50),
	@cHinweis NVARCHAR(255),
	@dMailVersand DATETIME,
	@dGedruckt DATETIME,
	@nFulfillment INT,
	@kLieferantenBestellung INT,
	@kSessionId INT,
	@kLieferschein INT OUTPUT
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @retry INT;
DECLARE @CreatedTransaction INT;
DECLARE @OldContextInfo VARBINARY(128);
DECLARE @Identities TABLE (kAutoInc INT IDENTITY(0, 1) NOT NULL, kLieferschein INT NOT NULL);

SET @kLieferschein = -1;

SET @retry = 5;

WHILE @retry > 0
BEGIN	
	IF(CONTEXT_INFO() IS NOT NULL)
		SET @OldContextInfo = CONTEXT_INFO();
	ELSE
		SET @OldContextInfo = 0x0;
	SET CONTEXT_INFO 0x5113;
	BEGIN TRY
		IF (@@TRANCOUNT = 0)
		BEGIN
		   SET @CreatedTransaction = 1;
		   BEGIN TRANSACTION
		END;
		ELSE
		BEGIN
			SET @CreatedTransaction = 0;
			SAVE TRANSACTION Savepoint0;
		END;
		IF(OBJECT_ID('tempdb..#Lieferschein') IS NOT NULL)
		BEGIN
			DROP TABLE #Lieferschein;
		END
		CREATE TABLE #Lieferschein(
	        kAutoInc INT IDENTITY(0, 1) NOT NULL,
			kBestellung INT,
			kBenutzer INT,
			cLieferscheinNr NVARCHAR(50),
			cHinweis NVARCHAR(255),
			dMailVersand DATETIME,
			dGedruckt DATETIME,
			nFulfillment INT,
			kLieferantenBestellung INT,
			kSessionId INT
		);
		IF(@xLieferschein IS NOT NULL)
		BEGIN
			INSERT INTO #Lieferschein(kBestellung, kBenutzer, cLieferscheinNr, cHinweis, dMailVersand,
				dGedruckt, nFulfillment, kLieferantenBestellung,kSessionId)
				SELECT Lieferschein.ID.value('kBestellung[1]', 'INT'),
						Lieferschein.ID.value('kBenutzer[1]', 'INT'),
						Lieferschein.ID.value('cLieferscheinNr[1]', 'NVARCHAR(50)'),
						Lieferschein.ID.value('cHinweis[1]', 'NVARCHAR(255)'),
						Lieferschein.ID.value('dMailVersand[1]', 'DATETIME'),
						Lieferschein.ID.value('dGedruckt[1]', 'DATETIME'),
						Lieferschein.ID.value('nFulfillment[1]', 'INT'),
						Lieferschein.ID.value('kLieferantenBestellung[1]', 'INT'),
						Lieferschein.ID.value('kSessionId[1]', 'INT')
					FROM @xLieferschein.nodes('Lieferschein') AS Lieferschein(ID);
		END
		ELSE
		BEGIN
			INSERT INTO #Lieferschein(kBestellung, kBenutzer, cLieferscheinNr, cHinweis, dMailVersand,
				dGedruckt, nFulfillment, kLieferantenBestellung, kSessionId)
				VALUES(@kBestellung, @kBenutzer, @cLieferscheinNr, @cHinweis, @dMailVersand, 
				@dGedruckt,@nFulfillment, @kLieferantenBestellung, @kSessionId);
		END	

		INSERT INTO dbo.tLieferschein(kBestellung, kBenutzer, cLieferscheinNr, cHinweis, dMailVersand, dErstellt,
									dGedruckt, nFulfillment, kLieferantenBestellung, kSessionId)
			OUTPUT inserted.kLieferschein INTO @Identities(kLieferschein)
			SELECT
				kBestellung, kBenutzer, cLieferscheinNr, cHinweis, dMailVersand, GETDATE(),
				dGedruckt, nFulfillment, kLieferantenBestellung, kSessionId
			FROM #Lieferschein;

		IF(OBJECT_ID('tempdb..#LieferscheinBestellungenConnectorUpload') IS NOT NULL)
		BEGIN
			DROP TABLE #LieferscheinBestellungenConnectorUpload;
		END
		CREATE TABLE #LieferscheinBestellungenConnectorUpload
		(
			kAuftrag INT
		);

		INSERT INTO #LieferscheinBestellungenConnectorUpload (kAuftrag)
			SELECT DISTINCT
				tAuftrag.kAuftrag
			FROM Verkauf.tAuftrag
			JOIN #Lieferschein ON #Lieferschein.kBestellung = tAuftrag.kAuftrag
			JOIN dbo.tShop ON tShop.kShop = tAuftrag.kShop
			WHERE tShop.nTyp = 1;

		IF EXISTS (SELECT 1 FROM #LieferscheinBestellungenConnectorUpload)
		BEGIN
			UPDATE Verkauf.tAuftrag
			SET tAuftrag.cInet = 'Y'
			FROM Verkauf.tAuftrag
			JOIN #LieferscheinBestellungenConnectorUpload ON #LieferscheinBestellungenConnectorUpload.kAuftrag = tAuftrag.kAuftrag;
		END;

		SELECT @kLieferschein = MAX(tIdentities.kLieferschein)
		  FROM @Identities AS tIdentities

		-- Kundenhistorie schreiben
		IF(EXISTS(SELECT tAuftrag.kKunde 
					FROM Verkauf.tAuftrag 
					JOIN #Lieferschein ON #Lieferschein.kBestellung = tAuftrag.kAuftrag
					AND tAuftrag.kKunde IS NOT NULL))
		BEGIN
			INSERT INTO Kunde.tHistorie (kKunde,
										kAuftrag,
										kVorgang,
										kKey,
										cValue1,
										cValue2,
										cValue3,
										dErstellt,
										kBenutzer)
				SELECT tAuftrag.kKunde,
					#Lieferschein.kBestellung,
					2,
					tIdentities.kLieferschein,
					#Lieferschein.cLieferscheinNr,
					tAuftrag.cAuftragsNr,
					NULL,
					GETDATE(),
					#Lieferschein.kBenutzer
				FROM #Lieferschein
				JOIN Verkauf.tAuftrag ON tAuftrag.kAuftrag = #Lieferschein.kBestellung
				JOIN @Identities AS tIdentities ON tIdentities.kAutoInc = #Lieferschein.kAutoInc
			   WHERE tAuftrag.kKunde IS NOT NULL;
		END
				
		DROP TABLE #Lieferschein;
		SET @retry = -1;
		SET CONTEXT_INFO @OldContextInfo;
		IF (@CreatedTransaction = 1)
		BEGIN
			COMMIT
		END
	END TRY
	BEGIN CATCH
	IF(ERROR_NUMBER() = 1205)
			BEGIN
				SET @retry = @retry - 1;
				IF(@CreatedTransaction = 1)
				BEGIN
				    ROLLBACK TRANSACTION;
				END
				ELSE
				BEGIN
				    ROLLBACK TRANSACTION Savepoint0;
				END
				IF(@retry = 0)
				BEGIN
					SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();
					RAISERROR (	@ErrorMessage, 
								@ErrorSeverity,
								@ErrorState
					);
					SET CONTEXT_INFO 0x0;
					RETURN;
				END
				WAITFOR DELAY '00:00:01';
			END
			ELSE 
			BEGIN
				SET @retry = -1;
				SELECT @ErrorMessage = ERROR_MESSAGE(), @ErrorSeverity = ERROR_SEVERITY(), @ErrorState = ERROR_STATE();

				IF(@CreatedTransaction = 1)
				BEGIN
				    ROLLBACK TRANSACTION;
				END
				ELSE
				BEGIN
				    ROLLBACK TRANSACTION Savepoint0;
				END
				RAISERROR (	@ErrorMessage, 
							@ErrorSeverity,
							@ErrorState
				);
				SET CONTEXT_INFO @OldContextInfo;
				RETURN;
			END
	END CATCH
	SET CONTEXT_INFO @OldContextInfo;	
END
