CREATE PROCEDURE [Gutschrift].[spGutschriftPosStuecklisteLoeschen]
	@kGutschriftStueckliste INT
AS
BEGIN
	DECLARE @xGutschriftPos AS XML;
	SET @xGutschriftPos = (
		SELECT dbo.tGutschriftPos.kGutschriftpos AS kGutschriftPos
			FROM dbo.tGutschriftPos 
			WHERE dbo.tGutschriftPos.kGutschriftStueckliste = @kGutschriftStueckliste
		FOR XML PATH('Gutschriftpos'), TYPE
	);
	EXEC Gutschrift.spGutschriftposLoeschen @xGutschriftpos = @xGutschriftpos;
END
