CREATE PROCEDURE [Streckenlager].[spUmlagerungAbschliessen]
@kUmlagerung INT, -- Die Umlagerung, die abgeschloßen werden soll
@kBenutzer INT ,
@cKommentar NVARCHAR(255)

-- Funktion: Alle Warenlagereingänge ein Umlagerung die nicht eingelagert wurden, werden ausgebucht
	
AS

DECLARE @kLieferantenBestellung INT
DECLARE @kBestellung INT


SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;



BEGIN TRY

	   SELECT @kLieferantenBestellung = kLieferantenBestellung, @kBestellung =  dbo.tUmlagerung.kBestellung
	   FROM dbo.tUmlagerung
	   WHERE kUmlagerung = @kUmlagerung;

	   IF(@kLieferantenBestellung > 0)
	   BEGIN

		   DECLARE @xWarenlagerAusgaenge XML;
		   SET @xWarenlagerAusgaenge = (
					SELECT dbo.tWarenLagerEingang.kWarenLagerEingang,
					0 AS kLieferscheinPos,
					dbo.tWarenLagerEingang.fAnzahlAktuell AS fAnzahl,
					@cKommentar AS cKommentar,
					@kBenutzer AS kBenutzer,
					30 AS kBuchungsart
					FROM dbo.tUmlagerung
					JOIN Verkauf.vAuftragPosition ON vAuftragPosition.kAuftrag = dbo.tUmlagerung.kBestellung
					JOIN dbo.tWarenLagerEingang ON dbo.tWarenLagerEingang.kBestellPosUmlagerung = vAuftragPosition.kAuftragPosition
					WHERE dbo.tUmlagerung.kUmlagerung = @kUmlagerung
					AND  dbo.tWarenLagerEingang.fAnzahlAktuell > 0

					FOR XML PATH('WarenAusgang'), TYPE
				);

			IF(@xWarenlagerAusgaenge IS NOT NULL)
			BEGIN

				EXEC dbo.spWarenlagerAusgangSchreiben
				@xWarenlagerAusgaenge = @xWarenlagerAusgaenge;

			END;

			EXEC [Lieferantenbestellung].[spLieferantenBestellungStatusAendern] 
			@kLieferantenBestellung =  @kLieferantenBestellung,
			@nStatus = 500;

		END;

		UPDATE dbo.tUmlagerung
		SET nStatus = 3
		WHERE kUmlagerung = @kUmlagerung;

		UPDATE Verkauf.tAuftrag SET tAuftrag.nKomplettAusgeliefert = 2
		WHERE kAuftrag = @kBestellung;

        DELETE FROM dbo.tPicklistePos WHERE nStatus <= 20 AND kBestellung = @kBestellung;

		DECLARE @Auftrag AS Verkauf.TYPE_spAuftragEckdatenBerechnen;
		INSERT INTO @Auftrag ( kAuftrag )
		VALUES ( @kBestellung);

		EXEC Verkauf.spAuftragEckdatenBerechnen @Auftrag = @Auftrag;

END TRY
BEGIN CATCH


    DECLARE @ErrorMessage NVARCHAR(4000);
    DECLARE @ErrorSeverity INT;
    DECLARE @ErrorState INT;

    SET @ErrorMessage =  ERROR_MESSAGE();
    SET @ErrorSeverity =  ERROR_SEVERITY();
    SET @ErrorState =  ERROR_STATE();

    RAISERROR (@ErrorMessage, 
		  @ErrorSeverity,
		  @ErrorState);
END CATCH
