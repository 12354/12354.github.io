CREATE PROCEDURE [dbo].[spRaiseError]
	@ErrorNumber INT,
	@ErrorData XML,
	@WithRollback BIT = 1
AS 
BEGIN
	DECLARE @kErrorId VARCHAR(100);
	DECLARE @kErrorMessage INT;
	SELECT @kErrorMessage = kErrorMessage 
		FROM dbo.tSqlErrorMessages
		WHERE kErrorMessage = @ErrorNumber;
	IF(@kErrorMessage IS NULL)
	BEGIN
		RAISERROR('Nicht referenzierter Fehler %s wurde geworfen.', 18,1, @ErrorNumber);
		RETURN
	END
	IF(@@TRANCOUNT > 0 AND @WithRollback = 1)
	BEGIN
		ROLLBACK;
	END	
	INSERT INTO dbo.tSqlError 
	(
		kErrorMessage,
		cErrorData
	)
	VALUES 
	(
		@ErrorNumber,
		@ErrorData
	);
	SET @kErrorId = CAST(SCOPE_IDENTITY() AS VARCHAR(100));
	RAISERROR('SQLError %s', 18, 1, @kErrorId);
END
