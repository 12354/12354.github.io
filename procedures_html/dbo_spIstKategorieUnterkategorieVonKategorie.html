CREATE PROCEDURE [dbo].[spIstKategorieUnterkategorieVonKategorie]
    @potentialDescendant TYPE_spIstKategorieUnterkategorieVonKategorie READONLY,
	@kKategorie INT
AS
BEGIN
    SET NOCOUNT ON;

	SELECT CASE WHEN EXISTS
	( 
		SELECT 1
		FROM dbo.ifKategoriebaum(@kKategorie) 
		WHERE ifKategoriebaum.kKategorie IN 
		(
			SELECT kKategorien
			FROM @potentialDescendant
		) 
	) 
	THEN 1 
	ELSE 0 
	END
	
END
