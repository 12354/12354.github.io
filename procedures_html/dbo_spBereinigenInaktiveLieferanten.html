dbo.spBereinigenInaktiveLieferanten

Description: Die Procedure löscht alle inaktiven Lieferanten endgültig aus der Datenbank.

Long Description: Keine ausführliche Beschreibung hinterlegt.

Created At: 2024-01-12 03:32:03

Updated At: 2024-01-12 03:32:03

CREATE PROCEDURE [dbo].[spBereinigenInaktiveLieferanten]
AS
SET NOCOUNT OFF;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
BEGIN
    DELETE FROM dbo.tlieferant
    WHERE tlieferant.cAktiv = 'N';

    DELETE FROM dbo.tliefartikel
    WHERE tliefartikel.tLieferant_kLieferant NOT IN( SELECT tlieferant.kLieferant
                                                     FROM dbo.tlieferant );
END;
