dbo.spBildLoeschenWennNichtVerwendet

Description: Die Procedure löscht Bilder die nicht mehr verwendet werden.

Long Description: Keine ausführliche Beschreibung hinterlegt.

CREATE PROC [dbo].[spBildLoeschenWennNichtVerwendet]
@Bilder TYPE_spBildLoeschenWennNichtVerwendet READONLY
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
BEGIN
    IF(object_id('tempdb..#spBildLoeschenWennNichtVerwendet') IS NOT NULL)
    BEGIN
	   DROP TABLE #spBildLoeschenWennNichtVerwendet;
    END;

    CREATE TABLE #spBildLoeschenWennNichtVerwendet (kBild INT)
    INSERT INTO #spBildLoeschenWennNichtVerwendet(kBild)
	SELECT kBild FROM @Bilder

    IF((SELECT COUNT(1) FROM #spBildLoeschenWennNichtVerwendet) = 0)
    BEGIN
	   RETURN;
    END;

	DELETE #spBildLoeschenWennNichtVerwendet
	FROM #spBildLoeschenWennNichtVerwendet
	JOIN dbo.tArtikelbildPlattform ON #spBildLoeschenWennNichtVerwendet.kBild = tArtikelbildPlattform.kBild;

	DELETE #spBildLoeschenWennNichtVerwendet
	FROM #spBildLoeschenWennNichtVerwendet
	JOIN dbo.tEigenschaftWertPict ON #spBildLoeschenWennNichtVerwendet.kBild = tEigenschaftWertPict.kBild;

	DELETE #spBildLoeschenWennNichtVerwendet
	FROM #spBildLoeschenWennNichtVerwendet
	JOIN dbo.tMerkmalBildPlattform ON #spBildLoeschenWennNichtVerwendet.kBild = tMerkmalBildPlattform.kBild;

	DELETE #spBildLoeschenWennNichtVerwendet
	FROM #spBildLoeschenWennNichtVerwendet
	JOIN dbo.tMerkmalwertBildPlattform ON #spBildLoeschenWennNichtVerwendet.kBild = tMerkmalwertBildPlattform.kBild;

	DELETE #spBildLoeschenWennNichtVerwendet
	FROM #spBildLoeschenWennNichtVerwendet
	JOIN dbo.tKategoriebildPlattform ON #spBildLoeschenWennNichtVerwendet.kBild = tKategoriebildPlattform.kBild;

	DELETE #spBildLoeschenWennNichtVerwendet
	FROM #spBildLoeschenWennNichtVerwendet
	JOIN dbo.tHerstellerBildPlattform ON #spBildLoeschenWennNichtVerwendet.kBild = tHerstellerBildPlattform.kBild;	

    DELETE dbo.tBild
	FROM dbo.tbild
	JOIN #spBildLoeschenWennNichtVerwendet ON tBild.kBild = #spBildLoeschenWennNichtVerwendet.kBild;

END
