CREATE PROCEDURE [Amazon].[spAmazonPosLoeschen]
	@kAmazonBestellungPos INT
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @ErrorMessage NVARCHAR(4000);
DECLARE @ErrorSeverity INT;
DECLARE @ErrorState INT;
DECLARE @retry INT;
SET @retry = 5;
WHILE @retry > 0
BEGIN
	SET CONTEXT_INFO 0x5103;
	BEGIN TRY  
		DECLARE @typeArtikel AS TYPE_spUpdateLagerbestand;
		INSERT INTO @typeArtikel (kArtikel)
		SELECT DISTINCT tExterneReservierung.kArtikel
		FROM dbo.tExterneReservierung
		WHERE tExterneReservierung.kKey = @kAmazonBestellungPos
				AND tExterneReservierung.nTyp = 3;

        DECLARE @TranCount INT = @@TRANCOUNT;
        IF(@TranCount = 0)
        BEGIN
            BEGIN TRANSACTION  
        END

		DELETE FROM dbo.pf_amazon_bestellungpos
		WHERE kAmazonBestellungPos = @kAmazonBestellungPos;
		
		DELETE FROM dbo.tExterneReservierung
		WHERE kKey = @kAmazonBestellungPos AND tExterneReservierung.nTyp = 3;
		
		EXEC spUpdateLagerbestand @typeArtikel;
		
		SET @retry = 0;
		SET CONTEXT_INFO 0x0;
		IF(@TranCount = 0)
        BEGIN           
            COMMIT TRANSACTION      
        END
    END TRY       
    BEGIN CATCH           
        IF(ERROR_NUMBER() = 1205 AND @TranCount = 0 AND XACT_STATE() <> 0)           
        BEGIN              
            SET @retry = @retry - 1;               
            ROLLBACK       
        END          
        ELSE          
        BEGIN              
            SET @retry = 0;               
            THROW;           
        END      
    END CATCH
END
