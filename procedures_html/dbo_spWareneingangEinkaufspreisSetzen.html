CREATE PROCEDURE [dbo].[spWareneingangEinkaufspreisSetzen]
@kLieferantenBestellungPos INT,
@fEKNetto DECIMAL(25, 13),
@kWarenlagerEingang INT = 0
AS  
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;
DECLARE @OldContextInfo VARBINARY(128);
BEGIN TRANSACTION

	-- Context Info
	IF(CONTEXT_INFO() IS NOT NULL)
	BEGIN
		SET @OldContextInfo = CONTEXT_INFO();
	END
	ELSE
	BEGIN
		SET @OldContextInfo = 0x0;
	END
	DECLARE @hash VARBINARY(128);
	SELECT @hash = HASHBYTES('SHA1', 'dbo.spWareneingangEinkaufspreisSetzen');
	SET CONTEXT_INFO @hash; -- ContextInfo festlegen

	IF(@kWarenlagerEingang > 0)
	BEGIN

		UPDATE dbo.tWarenLagerEingang
		SET tWarenLagerEingang.fEKEinzel = @fEKNetto
		WHERE tWarenLagerEingang.kWarenLagerEingang = @kWarenlagerEingang;

	END;
	ELSE
	BEGIN

		UPDATE dbo.tWarenLagerEingang
		SET tWarenLagerEingang.fEKEinzel = @fEKNetto
		WHERE tWarenLagerEingang.kLieferantenBestellungPos = @kLieferantenBestellungPos;

	END;


	SET CONTEXT_INFO @OldContextInfo;

COMMIT
