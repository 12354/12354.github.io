CREATE PROCEDURE [Maintenance].[spGarbageCollection]
AS
SET NOCOUNT ON;
SET ANSI_NULLS ON;
SET ANSI_NULL_DFLT_ON ON;
SET ANSI_PADDING ON;
SET CONCAT_NULL_YIELDS_NULL ON;
SET XACT_ABORT OFF;

BEGIN
    IF OBJECT_ID('dbo.tOptions') IS NOT NULL
    BEGIN
        MERGE INTO dbo.tOptions
        USING (VALUES('Maintenance.LastGarbageCollection', CONVERT(CHAR(10), GETUTCDATE(), 120))) AS Source(cKey, cValue) ON tOptions.cKey = Source.cKey
        WHEN MATCHED AND tOptions.cValue < Source.cValue THEN
            UPDATE SET tOptions.cValue = Source.cValue
        WHEN NOT MATCHED BY TARGET THEN
            INSERT (cKey, cValue) VALUES (Source.cKey, Source.cValue);
        IF @@ROWCOUNT > 0
        BEGIN
            -- Garbage Collection beginnen
            DECLARE @spName NVARCHAR(255);
            DECLARE SpCursor CURSOR FAST_FORWARD FOR 
                SELECT QUOTENAME(SCHEMA_NAME(procedures.schema_id))+'.'+QUOTENAME(procedures.name) FROM sys.procedures WHERE SCHEMA_NAME(procedures.schema_id) = 'Maintenance' AND name LIKE 'spGarbageCollection_%';
            OPEN SpCursor;
            FETCH NEXT FROM SpCursor INTO @spName;
            WHILE @@FETCH_STATUS = 0
            BEGIN
                EXEC @spName;
                FETCH NEXT FROM SpCursor INTO @spName;
            END
            CLOSE SpCursor;
            DEALLOCATE SpCursor;
        END
    END
END
