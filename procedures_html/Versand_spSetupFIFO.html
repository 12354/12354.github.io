CREATE PROCEDURE Versand.spSetupFIFO
AS
BEGIN

	DECLARE @FIFOAktiv CHAR(1);
	DECLARE @SQL NVARCHAR(MAX);

	SELECT
		@FIFOAktiv = tOptions.cValue
	FROM
		dbo.tOptions
	WHERE
		tOptions.cKey = 'FIFOAktiv'
	;

	IF (ISNULL(@FIFOAktiv, '0') = '0') -- FIFO soll inaktiv sein
	BEGIN

		SELECT @SQL = OBJECT_DEFINITION(OBJECT_ID('VersandIntern.vBestellPosOffenNonFIFO'));

		SET @SQL = REPLACE(@SQL, 'CREATE', 'ALTER');

		SET @SQL = REPLACE(@SQL, 'vBestellPosOffenNonFIFO', 'vBestellPosOffen');

		EXECUTE sys.sp_executesql @SQL;

		SELECT @SQL = OBJECT_DEFINITION(OBJECT_ID('[VersandIntern].[vBestellungLieferInfoProLagerNonFIFO]'));
		
		SET @SQL = REPLACE(@SQL, 'CREATE', 'ALTER');

		SET @SQL = REPLACE(@SQL, 'vBestellungLieferInfoProLagerNonFIFO', 'vBestellungLieferInfoProLager');

		EXECUTE sys.sp_executesql @SQL;
		
	END
	ELSE
	BEGIN
		
		SELECT @SQL = OBJECT_DEFINITION(OBJECT_ID('VersandIntern.vBestellPosOffenFIFO2012'));

		SET @SQL = REPLACE(@SQL, 'CREATE', 'ALTER');

		SET @SQL = REPLACE(@SQL, 'vBestellPosOffenFIFO2012', 'vBestellPosOffen');

		EXECUTE sys.sp_executesql @SQL;
		
		SELECT @SQL = OBJECT_DEFINITION(OBJECT_ID('[VersandIntern].[vBestellungLieferInfoProLagerFIFO2012]'));
		
		SET @SQL = REPLACE(@SQL, 'CREATE', 'ALTER');

		SET @SQL = REPLACE(@SQL, 'vBestellungLieferInfoProLagerFIFO2012', 'vBestellungLieferInfoProLager');

		EXECUTE sys.sp_executesql @SQL;

	END

END;
