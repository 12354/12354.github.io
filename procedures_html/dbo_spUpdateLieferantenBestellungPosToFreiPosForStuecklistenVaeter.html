CREATE PROCEDURE [dbo].[spUpdateLieferantenBestellungPosToFreiPosForStuecklistenVaeter]
	@StuecklistenVaterArtikelKeys TYPE_spUpdateLieferantenBestellungPosToFreiPosForStuecklistenVaeter READONLY 
AS
BEGIN

	DECLARE @ContexInfo VARBINARY(128)
	SELECT @ContexInfo = HASHBYTES('SHA1', 'spUpdateLieferantenBestellungPosToFreiPosForStuecklistenVaeter');
	SET CONTEXT_INFO @ContexInfo 

	BEGIN TRANSACTION UpdateLieferantenBestellungPos;
		
	BEGIN TRY

		DECLARE @updateLagerbestand dbo.TYPE_spUpdateLagerbestand
		INSERT INTO @updateLagerbestand (kArtikel) 
		SELECT kArtikel 
		FROM @StuecklistenVaterArtikelKeys

		UPDATE dbo.tLieferantenBestellungPos
		SET tLieferantenBestellungPos.nPosTyp = 0, tLieferantenBestellungPos.kArtikel = 0
		FROM dbo.tLieferantenBestellungPos
		WHERE tLieferantenBestellungPos.kArtikel IN 
		(
			SELECT kArtikel 
			FROM @StuecklistenVaterArtikelKeys
		)
		AND NOT EXISTS
		(
			SELECT 1 
			FROM @StuecklistenVaterArtikelKeys StuecklistenVaterArtikelKeys
			LEFT JOIN dbo.tArtikel ON StuecklistenVaterArtikelKeys.kArtikel = tArtikel.kArtikel
			WHERE tArtikel.kArtikel IS NULL
		)
		AND NOT EXISTS
		(
			SELECT 1 
			FROM @StuecklistenVaterArtikelKeys StuecklistenVaterArtikelKeys
			JOIN dbo.tArtikel ON StuecklistenVaterArtikelKeys.kArtikel = tArtikel.kArtikel
			WHERE tArtikel.kStueckliste <= 0
		)
		
		COMMIT TRANSACTION UpdateLieferantenBestellungPos;

		EXEC dbo.spUpdateLagerbestand @updateLagerbestand

	END TRY
	BEGIN CATCH

		DECLARE @ErrorData XML;
		SET @ErrorData =
			(
					SELECT ERROR_NUMBER() AS ErrorId, ERROR_MESSAGE()
					FOR 
						XML PATH('Error')
			)
		EXEC dbo.spRaiseError 13, @ErrorData, 1

	END CATCH

END
