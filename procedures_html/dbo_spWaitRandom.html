dbo.spWaitRandom

Description: Die Stored Procedure wird intern verwendet um zufällige Wartezeiten zu erzeugen. Sie kommt im Deadlockfall zum Einsatz.

CREATE PROCEDURE dbo.spWaitRandom (@SekundenDelay INT = 5)
AS
BEGIN
	DECLARE @Delay VARCHAR(20);
	SET @Delay = '00:00:' + CAST(CAST(@SekundenDelay * RAND() AS INT) AS NVARCHAR) +':' + CAST(CAST(1000*RAND() AS INT) AS NVARCHAR);
	WAITFOR DELAY @Delay;
END
