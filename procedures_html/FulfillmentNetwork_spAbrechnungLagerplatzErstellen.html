CREATE PROCEDURE [FulfillmentNetwork].[spAbrechnungLagerplatzErstellen]
    @kLieferant AS INT ,
    @kWarenlager AS INT ,
    @kBenutzer AS INT ,
    @Daten AS TYPE_spAbrechnungLagerplatzErstellen READONLY ,
    @nPreview TINYINT = 0 ,
    @cSessionId AS [NVARCHAR](255)
AS
    DECLARE @MODUS_PLATZ INT = 1;
    DECLARE @MODUS_TYPBEREICH INT = 2;
    DECLARE @MODUS_TYP INT = 3;
    DECLARE @MODUS_BEREICH INT = 4;
    DECLARE @MODUS_LAGER INT = 5;

    DECLARE @MODUS AS INT = 0;

    IF ( @kWarenlager = 0 )
        BEGIN
            DECLARE @cError NVARCHAR(255) = 'Es wurde kein kWarenlager übergeben';
            RAISERROR(@cError, 11, 1);		
            RETURN -1;
        END;
    IF @nPreview = 0
        BEGIN
            DELETE  FROM FulfillmentNetwork.tAbrechnungLagerplatz
            WHERE   kLieferant = @kLieferant
                    AND kWarenlager = @kWarenlager;

            DELETE  FROM FulfillmentNetwork.tAbrechnungLagerplatzPreview
            WHERE   cSessionId = @cSessionId
                    OR dZeitstempel < DATEADD(DAY, -2, GETDATE());

            INSERT  INTO FulfillmentNetwork.tAbrechnungLagerplatz
                    SELECT  @kLieferant AS kLieferant ,
                            kWarenLagerPlatz ,
                            kWarenLagerPlatzTyp ,
                            kWMSLagerBereich ,
                            @kWarenlager AS kWarenlager ,
                            CASE WHEN ( kWarenLagerPlatz > 0 )
                                 THEN @MODUS_PLATZ
                                 WHEN kWarenLagerPlatzTyp > 0
                                      AND kWMSLagerBereich > 0
                                 THEN @MODUS_TYPBEREICH
                                 WHEN kWarenLagerPlatzTyp > 0 THEN @MODUS_TYP
                                 WHEN kWMSLagerBereich > 0 THEN @MODUS_BEREICH
                                 ELSE @MODUS_LAGER
                            END AS nModus ,
                            nLagerkostenBerechnung ,
                            kArtikel ,
                            @kBenutzer AS kBenutzer
                    FROM    @Daten;
        END;
    ELSE
        BEGIN
            DELETE  FROM FulfillmentNetwork.tAbrechnungLagerplatzPreview
            WHERE   ( kLieferant = @kLieferant
                      AND kWarenlager = @kWarenlager
                    )
                    OR dZeitstempel < DATEADD(DAY, -2, GETDATE());

            INSERT  INTO FulfillmentNetwork.tAbrechnungLagerplatzPreview
                    SELECT  @kLieferant AS kLieferant ,
                            kWarenLagerPlatz ,
                            kWarenLagerPlatzTyp ,
                            kWMSLagerBereich ,
                            @kWarenlager AS kWarenlager ,
                            CASE WHEN ( kWarenLagerPlatz > 0 )
                                 THEN @MODUS_PLATZ
                                 WHEN kWarenLagerPlatzTyp > 0
                                      AND kWMSLagerBereich > 0
                                 THEN @MODUS_TYPBEREICH
                                 WHEN kWarenLagerPlatzTyp > 0 THEN @MODUS_TYP
                                 WHEN kWMSLagerBereich > 0 THEN @MODUS_BEREICH
                                 ELSE @MODUS_LAGER
                            END AS nModus ,
                            nLagerkostenBerechnung ,
                            kArtikel ,
                            @kBenutzer AS kBenutzer ,
                            @cSessionId AS cSessionId ,
                            GETDATE() AS dZeitstempel
                    FROM    @Daten;
		
        END;
