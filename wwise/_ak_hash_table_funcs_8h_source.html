33  // returns how far away the current slot is from the key's ideal position
34  template<typename KeyType>
35  AkInt32 DistanceFromIdealPos(AkInt32 iSlot, KeyType uKey, AkInt32 iEntriesMask)
36  {
37  // subtraction against an unmasked key and then masking afterwards,
38  // will give same result as if we masked the slot and key individually, first
39  // (also wraparound of the slot relative to the ukey also washes away with the masking)
40  return (iSlot - AkInt32(uKey)) & iEntriesMask;
41  }

43  // Internal helper function for AkHashTableBase; don't call this directly, use AK::HashTable::Get* functions instead
44  template<typename KeyType>
45  inline AkInt32 LinearProbe(const KeyType* pKeys, const bool* pbSlotOccupied, KeyType uKey, AkInt32 iSlot, AkUInt32 uNumEntries)
46  {
47  AkInt32 iDistFromBestSlot = 0;
48  AkInt32 iEntriesMask = (AkInt32)uNumEntries - 1;
49  for (; ; )
50  {
51  if (!pbSlotOccupied[iSlot])
52  return -1;
53  KeyType keyInSlot = pKeys[iSlot];
54  if (keyInSlot == uKey)
55  break;
56  if (iDistFromBestSlot > DistanceFromIdealPos(iSlot, keyInSlot, iEntriesMask))
57  return -1;
58  iSlot = (iSlot + 1) & iEntriesMask;
59  ++iDistFromBestSlot;
60  }
61  return iSlot;
62  }

64  // Internal helper function for AkHashTableBase; don't call this directly, use AkHashTableGet* functions instead
65  inline AkInt32 OccupiedProbe(const bool* pbSlotOccupied, AkInt32 iSlot, AkInt32 iNumEntries)
66  {
67  while (iSlot < iNumEntries)
68  {
69  // 64-bit load to scan 8 pbSlotOccupieds at once
70  // (safe to load a bit past the end of slotOccupied region because we cap against iNumEntries anyway)
71  if (AkUInt64 slotOccCompressed = *((AkUInt64*)(pbSlotOccupied + iSlot)))
72  {
73  iSlot = iSlot + AKPLATFORM::AkBitScanForward64(slotOccCompressed) / 8;
74  iSlot = (iSlot >= iNumEntries) ? -1 : iSlot;
75  return iSlot;
76  }
77  else
78  {
79  iSlot += 8;
80  }
81  }
82  return -1;
83  }

130  // runs the provided function over every active slot in the hashtable
131  // functype(AkUInt32 in_slot)
132  template<typename KeyType, typename FuncType>
133  inline void ForEachSlot(const AkHashTableBase<KeyType>* in_pHashTable, FuncType in_func)
134  {
135  AkUInt32 uNumReservedEntries = in_pHashTable->uNumReservedEntries;
136  bool* pbSlotOccupied = in_pHashTable->pbSlotOccupied;
137  for (AkUInt32 uBaseSlot = 0; uBaseSlot < uNumReservedEntries; uBaseSlot += 8)
138  {
139  AkUInt64 slotOccMask = *(AkUInt64*)(pbSlotOccupied + uBaseSlot);
140  while (slotOccMask)
141  {
142  AkUInt32 slotSubIdx = AKPLATFORM::AkBitScanReverse64(slotOccMask);
143  in_func(uBaseSlot + 7 - (slotSubIdx / 8));
144  slotOccMask ^= (0x8000000000000000ULL >> slotSubIdx);
145  }
146  }
147  }
