138 /// Initialization settings for the default implementation of the Memory Manager. For more details, see \ref memorymanager_init.
139 /// \sa AK::MemoryMgr
140 struct AkMemSettings
141 {
142  /// @name High-level memory allocation hooks. When not NULL, redirect allocations normally forwarded to rpmalloc.
143  //@{
144  AkMemInitForThread pfInitForThread; ///< (Optional) Thread-specific allocator initialization hook.
145  AkMemTermForThread pfTermForThread; ///< (Optional) Thread-specific allocator termination hook.
146  AkMemTrimForThread pfTrimForThread; ///< (Optional) Thread-specific allocator "trimming" hook. Used to relinquish memory resources when threads enter a period of inactivity.
147  AkMemMalloc pfMalloc; ///< (Optional) Memory allocation hook.
148  AkMemMalign pfMalign; ///< (Optional) Memory allocation hook.
149  AkMemRealloc pfRealloc; ///< (Optional) Memory allocation hook.
150  AkMemReallocAligned pfReallocAligned; ///< (Optional) Memory allocation hook.
151  AkMemFree pfFree; ///< (Optional) Memory allocation hook.
152  AkMemTotalReservedMemorySize pfTotalReservedMemorySize; ///< (Optional) Memory allocation statistics hook.
153  AkMemSizeOfMemory pfSizeOfMemory; ///< (Optional) Memory allocation statistics hook.
154  //@}
155 
156  /// @name Configuration.
157  //@{
158  AkUInt64 uMemAllocationSizeLimit; ///< When non-zero, limits the total amount of virtual and device memory allocated by AK::MemoryMgr.
159  bool bEnableSeparateDeviceHeap; ///< Enable use of device memory heap for all allocations (on applicable platforms).
160  AK::TempAlloc::InitSettings tempAllocSettings[AK::TempAlloc::Type_NUM]; ///< Configuration of behavior for the temporary-memory pools.
161  //@}
162 
163  /// @name Page allocation hooks, used by rpmalloc. Default to AKPLATFORM::AllocVM et al.
164  //@{
165  AkMemAllocVM pfAllocVM; ///< Virtual page allocation hook.
166  AkMemFreeVM pfFreeVM; ///< Virtual page allocation hook.
167  AkMemAllocVM pfAllocDevice; ///< Device page allocation hook.
168  AkMemFreeVM pfFreeDevice; ///< Device page allocation hook.
169  AkUInt32 uVMPageSize; ///< Virtual memory page size. Defaults to 0 which means auto-detect.
170  AkUInt32 uDevicePageSize; ///< Device memory page size. Defaults to 0 which means auto-detect.
171  AkUInt32 uMaxThreadLocalHeapAllocSize; ///< All memory allocations of sizes larger than this value will go to a global heap shared across all threads. Defaults to 0 which means all allocations go to a global heap.
172  //@}
173 
174  /// @name Memory allocation debugging.
175  //@{
176  AkMemDebugMalloc pfDebugMalloc; ///< (Optional) Memory allocation debugging hook. Used for tracking calls to pfMalloc.
177  AkMemDebugMalign pfDebugMalign; ///< (Optional) Memory allocation debugging hook. Used for tracking calls to pfMalign.
178  AkMemDebugRealloc pfDebugRealloc; ///< (Optional) Memory allocation debugging hook. Used for tracking calls to pfRealloc.
179  AkMemDebugReallocAligned pfDebugReallocAligned; ///< (Optional) Memory allocation debugging hook. Used for tracking calls to pfReallocAligned.
180  AkMemDebugFree pfDebugFree; ///< (Optional) Memory allocation debugging hook. Used for tracking calls to pfFree.
181  AkUInt32 uMemoryDebugLevel; ///< Default 0 disabled. 1 debug enabled. 2 stomp allocator enabled. 3 stomp allocator and debug enabled. User implementations may use multiple non-zero values to offer different features.
182  //@}
183 };
184 //@}
