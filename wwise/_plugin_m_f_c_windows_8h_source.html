53  /**
54  * \brief Initializes MFC for this plug-in.
55  *
56  * By default, an MFC-based DLL initializes itself when the DLL is loaded. However, not all plug-ins in a DLL
57  * container require MFC, so this makes sure to initialize MFC only on instantiation of a particular plug-in.
58  *
59  * If you need to modify the CWinApp class on your own, you can update the template.
60  *
61  * \code
62  * : public AK::Wwise::Plugin::PluginMFCWindows<MyWinApp>
63  * \endcode
64  *
65  * \aknote This should be the first interface of the plug-in, as it needs to be initialized before anything. \endaknote
66  */
67  template<typename CWinApp = ::CWinApp>
68  class PluginMFCWindows
69  {
70  /**
71  * \brief DLL singleton for MFC's CWinApp.
72  *
73  * This singleton is unique in the DLL. It should be initialized in the main app thread.
74  * Please note it is desired behavior to have multiple instances of this singleton in an application. However,
75  * MFC expects to have a single copy per DLL.
76  */
77  static std::unique_ptr<CWinApp> g_theApp;
78 
79  public:
80  /**
81  * \brief Initializes g_theApp singleton
82  */
83  PluginMFCWindows()
84  {
85  if (!g_theApp)
86  {
87  AFX_MANAGE_STATE(AfxGetStaticModuleState());
88  g_theApp.reset(new CWinApp);
89 
90  g_theApp->InitInstance();
91  }
92  }
93 
94  /**
95  * \brief Retrieves the CWinApp singleton.
96  *
97  * \return CWinApp*
98  */
99  static CWinApp* GetMFCApp()
100  {
101  return g_theApp;
102  }
103  };
