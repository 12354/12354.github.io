39 struct BasePlatformID
40 {
41  BasePlatformID():guid( GUID_NULL ){}
42 
43  // The Create function is a functionnal replacement for the BasePlatformID() constructor taking a GUID in parameters.
44  // This is to avoid situations where someone would pass a GUID instead of a BasePlatformID in a function call while migrating, which would
45  // compile but would lead the user into serious errors.
46  static BasePlatformID Create( const GUID& in_BasePlatformID )
47  {
48  BasePlatformID ret;
49  ret.guid = in_BasePlatformID;
50  return ret;
51  }
52 
53  static BasePlatformID Empty()
54  {
55  return BasePlatformID::Create(GUID_NULL);
56  }
57 
58  bool IsNull() const
59  {
60  return (guid == GUID_NULL) ? true : false;
61  }
62 
63  bool operator == ( const BasePlatformID& in_rOther ) const
64  {
65  return (in_rOther.guid == guid)? true : false;
66  }
67  bool operator != ( const BasePlatformID& in_rOther ) const
68  {
69  return (in_rOther.guid != guid)? true : false;
70  }
71  GUID guid;
72 
73  struct GUIDLessNative
74  {
75  bool operator()( const BasePlatformID& in_rLeft, const BasePlatformID& in_rRight ) const
76  {
77  return ( ::memcmp( &in_rLeft, &in_rRight, sizeof(BasePlatformID) ) < 0 );
78  }
79  };
80 };
