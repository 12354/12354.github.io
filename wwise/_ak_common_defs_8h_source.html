178 // Build a 32 bit class identifier based on the Plug-in type,
179 // CompanyID and PluginID.
180 //
181 // Parameters:
182 // - in_pluginType: A value from enum AkPluginType (4 bits)
183 // - in_companyID: CompanyID as defined in the Plug-in's XML file (12 bits)
184 // * 0-63: Reserved for Audiokinetic
185 // * 64-255: Reserved for clients' in-house Plug-ins
186 // * 256-4095: Assigned by Audiokinetic to third-party plug-in developers
187 // - in_pluginID: PluginID as defined in the Plug-in's XML file (16 bits)
188 // * 0-32767: Set freely by the Plug-in developer
189 #define AKMAKECLASSID( in_pluginType, in_companyID, in_pluginID ) \
190  ( (in_pluginType) + ( (in_companyID) << 4 ) + ( (in_pluginID) << ( 4 + 12 ) ) )
191

193 #define AKGETCOMPANYIDFROMCLASSID( in_classID ) ( ( (in_classID) & 0x0000FFF0 ) >> 4 )
194 #define AKGETPLUGINIDFROMCLASSID( in_classID ) ( ( (in_classID) & 0xFFFF0000 ) >> ( 4 + 12 ) )

544  /// Convert a channel, identified by a single channel bit, to a buffer index used in GetChannel() below, for a given channel config.
545  /// Standard indexing follows channel bit order (see AkSpeakerConfig.h). Pipeline/buffer indexing is the same but the LFE is moved to the end.
546  static inline AkUInt32 StandardToPipelineIndex(
547  AkChannelConfig in_channelConfig, ///< Channel configuration.
548  AkUInt32 in_uChannelIdx ///< Channel index in standard ordering to be converted to pipeline ordering.
549  )
550  {
551  if ( in_channelConfig.HasLFE() )
552  {
553  AKASSERT( in_channelConfig.eConfigType == AK_ChannelConfigType_Standard ); // in_channelConfig.HasLFE() would not have returned true otherwise.
554  AKASSERT( AK::GetNumNonZeroBits( in_channelConfig.uChannelMask ) );
555  AkUInt32 uIdxLFE = AK::GetNumNonZeroBits( ( AK_SPEAKER_LOW_FREQUENCY - 1 ) & in_channelConfig.uChannelMask );
556  if ( in_uChannelIdx == uIdxLFE )
557  return in_channelConfig.uNumChannels - 1;
558  else if ( in_uChannelIdx > uIdxLFE )
559  return in_uChannelIdx - 1;
560  }
561 
562  return in_uChannelIdx;
563  }

570  /// \return Address of the buffer of the ith channel.
571  /// \sa

573  /// - \ref fx_audiobuffer_struct_channels
574  inline AkSampleType * GetChannel(

594  /// Can be used to transform an incomplete into a complete buffer with valid data.
595  /// The invalid frames are made valid (zeroed out) for all channels and the validFrames count will be made equal to uMaxFrames.
596  void ZeroPadToMaxFrames()
597  {
598  AKASSERT(pData != nullptr || MaxFrames() == 0);
599  // The following members MUST be copied locally due to multi-core calls to this function.
600  const AkUInt32 uNumChannels = NumChannels();
601  const AkUInt32 uNumCurrentFrames = AkMin(uValidFrames, MaxFrames());
602  const AkUInt32 uNumZeroFrames = MaxFrames() - uNumCurrentFrames;
603  if ( uNumZeroFrames )
604  {
605  AKASSERT(pData != nullptr);
606  for ( AkUInt32 i = 0; i < uNumChannels; ++i )
607  {
608  AKPLATFORM::AkMemSet( GetChannel(i) + uNumCurrentFrames, 0, uNumZeroFrames * sizeof(AkSampleType) );
609  }
610  uValidFrames = MaxFrames();
611  }
612  }

621  {
622  uMaxFrames = 0;
623  uValidFrames = 0;
624  channelConfig.Clear();
625  void * pDataOld = pData;
626  pData = NULL;
