215 // Helpers.
216 inline void AK_SPEAKER_SETUP_FIX_LEFT_TO_CENTER( AkUInt32 &io_uChannelMask )
217 {
218  if( !(io_uChannelMask & AK_SPEAKER_FRONT_CENTER)
219  && !(io_uChannelMask & AK_SPEAKER_FRONT_RIGHT)
220  && (io_uChannelMask & AK_SPEAKER_FRONT_LEFT) )
221  {
222  io_uChannelMask &= ~AK_SPEAKER_FRONT_LEFT; // remove left
223  io_uChannelMask |= AK_SPEAKER_FRONT_CENTER; // add center
224  }
225 }
226 
227 inline void AK_SPEAKER_SETUP_FIX_REAR_TO_SIDE( AkUInt32 &io_uChannelMask )
228 {
229  if( io_uChannelMask & ( AK_SPEAKER_BACK_LEFT ) && !( io_uChannelMask & AK_SPEAKER_SIDE_LEFT ) )
230  {
231  io_uChannelMask &= ~( AK_SPEAKER_BACK_LEFT | AK_SPEAKER_BACK_RIGHT ); // remove rears
232  io_uChannelMask |= ( AK_SPEAKER_SIDE_LEFT | AK_SPEAKER_SIDE_RIGHT ); // add sides
233  }
234 }
235 
236 inline void AK_SPEAKER_SETUP_CONVERT_TO_SUPPORTED( AkUInt32 &io_uChannelMask )
237 {
238  AK_SPEAKER_SETUP_FIX_LEFT_TO_CENTER( io_uChannelMask );
239  AK_SPEAKER_SETUP_FIX_REAR_TO_SIDE( io_uChannelMask );
240 }

401 /// Convert channel indices as they are ordered in standard (WAV) or Wwise sound engine (WEM) wave files
402 /// (which follow channel mask bit values, except that the LFE is at the end in the case of WEMs)
403 /// into display indices. Desired display order is L-R-C-SL-SR-RL-RR-HL-HR-HC-HRL-HRR-HRC-T-LFE. Note that 4-5.x configurations
404 /// may define back or side channels. Either way they are "Surround" channels and are assigned to "SL, SR" names.
405 static inline unsigned int StdChannelIndexToDisplayIndex( AkChannelOrdering in_eOrdering, unsigned int in_uChannelMask, unsigned int in_uChannelIdx )
406 {
407  if ( in_eOrdering == ChannelOrdering_Standard )
408  {
409  unsigned int uNumChannelsFront = ChannelMaskToNumChannels( in_uChannelMask & AK_SPEAKER_SETUP_FRONT );
410  if ( ( in_uChannelMask & AK_SPEAKER_LOW_FREQUENCY )
411  && ( in_uChannelIdx == uNumChannelsFront ) )
412  {
413  // Lfe. Return penultimate channel.
414  in_uChannelIdx = ChannelMaskToNumChannels( in_uChannelMask ) - 1;
415  }
416  else if ( in_uChannelIdx >= uNumChannelsFront ) // strictly greater than uNumChannelsFront (lfe index) if lfe is present, greater or equal otherwise.
417  {
418  // Back channel. Return index or index-1 if there is an LFE (uLfeOffset==1).
419  unsigned int uLfeOffset = ( in_uChannelMask & AK_SPEAKER_LOW_FREQUENCY ) ? 1 : 0;
420 
421  // 6-7.x: Need to swap back and sides.
422  if ( HasSideAndRearChannels( in_uChannelMask ) )
423  {
424  unsigned int uRearIdx = uNumChannelsFront + uLfeOffset;
425  unsigned int uSideIdx = uRearIdx + 2;
426  unsigned int uAfterSideIdx = uSideIdx + 2;
427  if ( in_uChannelIdx < uAfterSideIdx )
428  {
429  if ( in_uChannelIdx >= uSideIdx )
430  in_uChannelIdx -= 2; // input is side, swap it with back.
431  else
432  in_uChannelIdx += 2; // input is back, swap it with side.
433  }
434  }
435  in_uChannelIdx -= uLfeOffset; // compensate for LFE if it was skipped above.
436  }
437  }
438  else
439  {
440  // 6-7.x: Need to swap back and sides.
441  if ( HasSideAndRearChannels( in_uChannelMask ) )
442  {
443  unsigned int uRearIdx = ChannelMaskToNumChannels( in_uChannelMask & AK_SPEAKER_SETUP_FRONT );
444  unsigned int uMaxIdx = uRearIdx + 4; // Side and rear channels.
445 
446  if ( in_uChannelIdx >= uRearIdx
447  && in_uChannelIdx < uMaxIdx )
448  {
449  // Surround channel (not LFE).
450  unsigned int uSideIdx = uRearIdx + 2;
451  if ( in_uChannelIdx >= uSideIdx )
452  in_uChannelIdx -= 2; // input is side, swap it with back.
453  else
454  in_uChannelIdx += 2; // input is back, swap it with side.
455  }
456  }
457  }
458 
459  return in_uChannelIdx;
460 }
