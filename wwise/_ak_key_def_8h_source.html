50 // A helper struct that can be used as a T_ITEM in an CAkKeyArray, when you need to only preform a shallow copy
51 // on the data that you are referencing. For example if you wanted to to use an AkArray type.
52 // NOTE: AllocData() and FreeData() must be explicitly called, or else pData can be manually Alloc'd/Free'd.
53 template < typename T_KEY, typename T_DATA, class T_ALLOC = ArrayPoolDefault >
54 struct AkKeyDataPtrStruct: public T_ALLOC
55 {
56  AkKeyDataPtrStruct(): pData(NULL) {}
57  AkKeyDataPtrStruct(T_KEY in_key): key(in_key), pData(NULL) {}
58 
59  T_KEY key;
60  T_DATA* pData;
61  bool operator ==(const AkKeyDataPtrStruct<T_KEY,T_DATA>& in_Op) const
62  {
63  return ( key == in_Op.key );
64  }
65 
66  bool AllocData()
67  {
68  AKASSERT( !pData );
69  pData = (T_DATA*) T_ALLOC::Alloc( sizeof( T_DATA ));
70  if (pData)
71  {
72  AkPlacementNew( pData ) T_DATA();
73  return true;
74  }
75  return false;
76  }
77 
78  void FreeData()
79  {
80  if( pData )
81  {
82  pData->~T_DATA();
83  T_ALLOC::Free(pData);
84  pData = NULL;
85  }
86  }
87 
88  static AkForceInline T_KEY& Get( AkKeyDataPtrStruct<T_KEY, T_DATA>& in_val ) { return in_val.key; }
89 };
