67 //Single producer, single consumer pattern implementation.
68 class AkRingBuffer
69 {
70 public:
71  AkRingBuffer()
72  : m_nbItems(0)
73  , m_readIndex(0)
74  , m_writeIndex(0)
75  , m_nbReadableItems(0)
76  {
77  }
78 
79  AKRESULT Init(AkUInt32 nbItems)
80  {
81  m_nbItems = nbItems;
82  m_readIndex = 0;
83  m_writeIndex = 0;
84  m_nbReadableItems = 0;
85 
86  m_data = reinterpret_cast<T*>(TAlloc::Alloc(m_nbItems * sizeof(T)));
87  if (m_data == NULL)
88  {
89  return AK_InsufficientMemory;
90  }
91  return AK_Success;
92  }
93 
94  void Term()
95  {
96  m_nbItems = 0;
97  m_readIndex = 0;
98  m_writeIndex = 0;
99  m_nbReadableItems = 0;
100 
101  if (m_data != NULL)
102  {
103  TAlloc::Free(reinterpret_cast<void*>(m_data));
104  m_data = NULL;
105  }
106  }
107 
108  // Reset ringbuffer to initial state without freeing memory. Not thread-safe.
109  void Reset()
110  {
111  m_readIndex = 0;
112  m_writeIndex = 0;
113  m_nbReadableItems = 0;
114  }
115 
116  // ---- Producer ---- //
117 
118  AkUInt32 GetWriteIndex() const
119  {
120  return m_writeIndex;
121  }
122 
123  T* GetWritePtr()
124  {
125  return &m_data[m_writeIndex];
126  }
127 
128  void IncrementWriteIndex(AkUInt32 nbItems)
129  {
130  AKASSERT(GetNbWritableItems() >= nbItems);
131 
132  m_writeIndex = (m_writeIndex + nbItems) % m_nbItems;

136 
137  // ---- Consumer ----
138

148 
149  // Peek at any item between the read and write pointer without advancing the read pointer.
150  const T* Peek(AkUInt32 uOffset) const
151  {
152  AKASSERT((AkUInt32)m_nbReadableItems > uOffset);
153  AkUInt32 uReadIndex = (m_readIndex + uOffset) % m_nbItems;
154  return &m_data[uReadIndex];
155  }
156 
157  void IncrementReadIndex(AkUInt32 nbItems)
158  {
159  AKASSERT((AkUInt32)m_nbReadableItems >= nbItems);
160 
161  m_readIndex = (m_readIndex + nbItems) % m_nbItems;
162 
163  AkAtomicSub32(&m_nbReadableItems, nbItems);
164  }
165 
166  AkUInt32 GetNbReadableItems() const
167  {
168  return m_nbReadableItems;
169  }
170 
171  AkUInt32 GetNbWritableItems() const
172  {
173  return m_nbItems - (AkUInt32)m_nbReadableItems;
174  }
175 
176  AkUInt32 Size() const
177  {
178  return m_nbItems;
179  }
180 
181  // Warning: requires external locking to prevent concurrent Grow+Read in a multi-threaded scenario.
182  // Like the rest of the class, assumes a single writing thread.
183  bool Grow(AkUInt32 in_uGrowBy)
184  {
185  AkUInt32 uTargetItems = m_nbItems + in_uGrowBy;
186  if (T* pNewData = reinterpret_cast<T*>(TAlloc::Alloc(uTargetItems * sizeof(T))))
187  {
188  if (m_nbReadableItems)
189  {
190  if (m_readIndex >= m_writeIndex)
191  {
192  // insert new free space in the middle of the buffer.
193 
194  if (m_writeIndex)
195  memcpy(pNewData, m_data, sizeof(T) * m_writeIndex);
196 
197  memcpy(pNewData + m_readIndex + in_uGrowBy, m_data + m_readIndex, sizeof(T) * (m_nbItems - m_readIndex));
198  m_readIndex += in_uGrowBy;
199  }
200  else
201  {
202  // insert new free space at the end of the buffer.
203 
204  memcpy(pNewData + m_readIndex, m_data + m_readIndex, sizeof(T) * (AkUInt32)m_nbReadableItems);
205  }
206  }
207 
208  TAlloc::Free(reinterpret_cast<void*>(m_data));
209  m_data = pNewData;
210  m_nbItems = uTargetItems;
211 
212  return true;
213  }
214  else
215  {
216  return false;
217  }
218  }
219 
220 private:
221  T* m_data{nullptr};
222 
223  AkUInt32 m_nbItems;
224  AkUInt32 m_readIndex;
225  AkUInt32 m_writeIndex;
226  AkAtomic32 m_nbReadableItems;
227 };
