40 /// Interface for an open-addressed hash table (or key-value array).
41 /// Relatively low-cost (O(1)) lookup, add, and removal by key. Typical use is to add values by key,
42 /// and then using GetFirstSlotByKey (and GetNextSlotByKey, if keys are not guaranteed to be unique)
43 /// to get a Slot into the AkHashTableBase, and then indexing AkHashTableBase::pValues by the provided Slot.
44 ///
45 /// Note that the AkHashTableBase referred to is not inherently thread-safe, so appropriate protections
46 /// will be required if the same hashTable is shared across multiple plug-in instances.
47 ///
48 /// Some hash functions are provided in AkMurMurHash.h which can be used for key generation.
49 /// In particular, this can be very useful for storing data on audio objects, or audio object channels,
50 /// over time. Note that if indexing by object-channel is desired, GetObjectChannelHash is recommended
51 /// to mitigate key-collisions from multiple channels of data. However, Audio Object IDs themselves are
52 /// chaotic enough that they can be used directly.
53

93  {

126  // runs the provided function over every active slot in the hashtable, which is used to determine if the element should be removed
127  // Note that this can move many entries in the table around, so previous pointers to elements, etc, should be considered invalidated
128  // bool functype(AkUInt32 in_slot)
129  template<typename KeyType, typename FuncType>
130  void RemoveIf(AkHashTableBase<KeyType>* in_pHashTable, FuncType in_func)
131  {
132  AkUInt32 uNumReservedEntries = in_pHashTable->uNumReservedEntries;
133  bool* pbSlotOccupied = in_pHashTable->pbSlotOccupied;
134  AkUInt32 uSlot = 0;
135  while (uSlot < uNumReservedEntries)
136  {
137  if (pbSlotOccupied[uSlot] && in_func(uSlot))
138  {
139  // if slot is occupied, and the function confirmed removal, remove this slot, but don't advance uSlot
140  RemoveSlot(in_pHashTable, (AkInt32)uSlot);
141  }
142  else
143  {
144  uSlot++;
145  }
146  }
147  }
